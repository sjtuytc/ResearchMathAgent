function _getTip(){ if(!_tip) _tip=$("ov-tooltip"); return _tip; }
function showTip(el, html){
  const _tip=_getTip(); if(!_tip) return;
  _tip.innerHTML=html;
  _tip.style.display="block";
  const r=el.getBoundingClientRect();
  moveTipXY(r.right+6, r.top);
}
function moveTipXY(x,y){
  const _tip=_getTip(); if(!_tip||_tip.style.display==="none") return;
  const mw=_tip.offsetWidth||220, mh=_tip.offsetHeight||60;
  _tip.style.left=(x+mw>window.innerWidth ? window.innerWidth-mw-8 : x)+"px";
  _tip.style.top=(y+mh>window.innerHeight ? window.innerHeight-mh-8 : y)+"px";
}
function hideTip(){ const _tip=_getTip(); if(_tip) _tip.style.display="none"; }
document.addEventListener("mousemove", ev=>moveTipXY(ev.clientX+12, ev.clientY-6));
document.addEventListener("mouseleave", hideTip);
document.addEventListener("mouseover", ev=>{
  const el=ev.target.closest(".info-icon");
  if(el){
    const tip=el.getAttribute("data-tip")||el.title||"";
    if(tip) showTip(el, tip.replace(/\n/g,"
"));
  } else if(!ev.target.closest("#ov-tooltip")){
    hideTip();
  }
});

// ── Donut chart helpers ───────────────────────────────────────────────
function drawDonut(svgEl, slices, r, sw){
  if(!svgEl) return;
  svgEl.innerHTML="";
  const total=slices.reduce((s,x)=>s+x.pct,0);
  if(total<=0){
    const t=document.createElementNS("http://www.w3.org/2000/svg","text");
    t.setAttribute("x","50%"); t.setAttribute("y","54%");
    t.setAttribute("text-anchor","middle");
    t.setAttribute("fill","var(--muted)"); t.setAttribute("font-size","10");
    t.textContent="no data"; svgEl.appendChild(t); return;
  }
  const cx=50, cy=50, circ=2*Math.PI*r;
  let offset=0;
  const norm=slices.map(x=>({...x, pct:x.pct/total*100}));
  for(const s of norm){
    const dash=(s.pct/100)*circ, gap=circ-dash;
    const c=document.createElementNS("http://www.w3.org/2000/svg","circle");
    c.setAttribute("cx",cx); c.setAttribute("cy",cy); c.setAttribute("r",r);
    c.setAttribute("fill","none"); c.setAttribute("stroke",s.color);
    c.setAttribute("stroke-width",sw);
    c.setAttribute("stroke-dasharray",`${dash} ${gap}`);
    c.setAttribute("stroke-dashoffset",-offset*circ/100);
    c.setAttribute("transform",`rotate(-90 ${cx} ${cy})`);
    c.style.cursor="default";
    c.onmouseenter=(ev)=>{ showTip(ev.target,`${esc(s.label)}
${s.valStr||("$"+s.value.toFixed(4))} · ${s.pct.toFixed(1)}%`); };
    c.onmouseleave=hideTip;
    svgEl.appendChild(c);
    offset+=s.pct;
  }
}
function donutLegend(wrap, slices, total){
  if(!wrap) return;
  wrap.innerHTML=slices.filter(s=>s.pct>0).map(s=>`
    

      
      ${esc(s.label)}
      ${s.valStr||("$"+s.value.toFixed(4))}
      ${(total>0?s.pct/total*100:0).toFixed(1)}%
    
`).join("");
}

async function pdfRenderAll(doc, scale, wrap, withTextLayer=true){
  wrap.innerHTML="";
  const n=doc.numPages;
  for(let i=1;i<=n;i++){
    const page=await doc.getPage(i);
    const vp=page.getViewport({scale});
    const pageWrap=document.createElement("div");
    pageWrap.className="pdf-page-wrap";
    pageWrap.dataset.page=i;
    pageWrap.style.cssText=`width:${vp.width}px;height:${vp.height}px;box-shadow:0 2px 14px rgba(0,0,0,.55)`;
    const canvas=document.createElement("canvas");
    canvas.width=vp.width; canvas.height=vp.height;
    canvas.style.cssText="display:block;background:#fff";
    pageWrap.appendChild(canvas);
    wrap.appendChild(pageWrap);
    await page.render({canvasContext:canvas.getContext("2d"),viewport:vp}).promise;
    // text selection layer (non-blocking)
    if(withTextLayer && typeof pdfjsLib!=="undefined"){
      page.getTextContent().then(tc=>{
        const tl=document.createElement("div");
        tl.className="textLayer";
        tl.style.cssText=`width:${vp.width}px;height:${vp.height}px`;
        pageWrap.appendChild(tl);
        try{
          const task=pdfjsLib.renderTextLayer({textContent:tc,container:tl,viewport:vp,textDivs:[]});
          const p=task.promise||task;
          (p instanceof Promise?p:Promise.resolve()).then(()=>tl.classList.add("ready")).catch(()=>{});
        }catch(e){}
      }).catch(()=>{});
    }
  }
}

// Convert LaTeX structural commands to markdown so MathJax can typeset the math
function cleanLatexForDisplay(s){
  if(!s) return "";
  const cap = e => e.charAt(0).toUpperCase()+e.slice(1).toLowerCase();
  // Named theorem-like environments → bold labels
  s = s.replace(/\\begin\{(theorem|lemma|proposition|corollary|claim|conjecture)\}(\s*\[[^\]]*\])?/gi,
    (_, e, opt) => `\n\n**${cap(e)}${opt?' '+opt.trim():''}.**\n`);
  s = s.replace(/\\begin\{(definition|example|remark|question|problem|exercise)\}(\s*\[[^\]]*\])?/gi,
    (_, e, opt) => `\n\n**${cap(e)}${opt?' '+opt.trim():''}.**\n`);
  s = s.replace(/\\begin\{proof\}(\s*\[[^\]]*\])?/gi,
    (_, opt) => `\n\n*Proof${opt?' '+opt.trim():''}.*\n`);
  s = s.replace(/\\end\{(theorem|lemma|proposition|corollary|claim|conjecture|definition|example|remark|question|problem|exercise|proof)\}/gi, '\n\n');
  // Lists
  s = s.replace(/\\begin\{(enumerate|itemize)\}/gi, '\n');
  s = s.replace(/\\end\{(enumerate|itemize)\}/gi, '\n');
  s = s.replace(/\\item(\[[^\]]*\])?\s*/g, '\n- ');
  // Sections
  s = s.replace(/\\section\*?\{([^}]+)\}/g, '\n\n## $1\n\n');
  s = s.replace(/\\subsection\*?\{([^}]+)\}/g, '\n\n### $1\n\n');
  s = s.replace(/\\paragraph\*?\{([^}]+)\}\s*/g, '\n\n**$1**\n\n');
  // Text formatting
  s = s.replace(/\\textbf\{([^}]+)\}/g, '**$1**');
  s = s.replace(/\\(?:emph|textit)\{([^}]+)\}/g, '*$1*');
  s = s.replace(/\\textrm\{([^}]+)\}/g, '$1');
  // Strip structural noise
  s = s.replace(/\\label\{[^}]*\}/g, '');
  s = s.replace(/\\(?:maketitle|noindent|medskip|bigskip|smallskip|newpage|clearpage|hfill|vfill)\b/g, '');
  s = s.replace(/\\title\{[^}]+\}/g, '');
  s = s.replace(/\\author\{[^}]+\}/g, '');
  s = s.replace(/\\date\{[^}]+\}/g, '');
  s = s.replace(/\\input\{[^}]+\}/g, '');
  s = s.replace(/~?\\(?:cite|ref|eqref)(?:\[[^\]]*\])?\{[^}]*\}/g, '');
  s = s.replace(/\\protect\b/g, '');
  s = s.replace(/\\(?:newcommand|renewcommand)\{[^}]+\}(?:\[[^\]]*\])?\{[^}]*\}/g, '');
  s = s.replace(/\n{3,}/g, '\n\n').trim();
  return s;
}

// Render content: always use markdown+MathJax; strip structural LaTeX first
function renderContent(node, text, stream=false){
  node.dataset.raw = text||"";
  md(node, cleanLatexForDisplay(text)||"*(empty)*", stream);
}

function extractStatement(tex){
  let m=tex.match(/\\begin\{document\}([\s\S]*?)\\end\{document\}/);
  let body=m?m[1]:tex;
  body=body.replace(/\\maketitle/g,"").replace(/\\title\{[^}]*\}/g,"")
           .replace(/\\author\{[^}]*\}/g,"").replace(/\\date\{[^}]*\}/g,"")
           .replace(/\\input\{[^}]*\}/g,"");
  return body.trim();
}

function issRenderAllIssues(issues){
  let filtered=issues;
  if(_issFilterStatus) filtered=filtered.filter(i=>i.status===_issFilterStatus);
  if(_issFilterType)   filtered=filtered.filter(i=>i.issue_type===_issFilterType);
  const byQ={};
  for(const i of filtered){
    const q=i.problem_id||"?";
    (byQ[q]||(byQ[q]=[])).push(i);
  }
  const list=$("iss-list");
  if(!list) return;
  $("i-empty").style.display="none"; $("i-body").style.display="block";
  list.innerHTML="";
  if(!filtered.length){
    list.innerHTML='
    No issues match the current filter.
    ';
    return;
  }
  const open=filtered.filter(i=>i.status==="open").length;
  const p0=filtered.filter(i=>i.priority==="P0").length;
  const p1=filtered.filter(i=>i.priority==="P1").length;
  const summary=document.createElement("div");
  summary.style.cssText="padding:6px 0;margin-bottom:8px;font-size:11px;color:var(--muted);border-bottom:1px solid var(--border)";
  summary.innerHTML=`${filtered.length} issues · ${open} open · ${p0} P0 · ${p1} P1`;
  list.appendChild(summary);
  for(const [qid, qIssues] of Object.entries(byQ)){
    const hdr=document.createElement("div");
    hdr.style.cssText="font-size:10px;font-weight:700;color:var(--accent);text-transform:uppercase;letter-spacing:.06em;padding:8px 0 4px;cursor:pointer";

    hdr.onclick=()=>{ selectQuestion(qid,null,selectedDataset||"first_proof_1"); issSetLevel("question"); setTab("issue"); };
    list.appendChild(hdr);

      list.appendChild(_issRowEl(issue));
    }
  }
}

async function runIssueAgent(url, label){
  if(issAgentEs){ issAgentEs.close(); }
  issAgentStart(label);
  issAgentLogAppend(`[${label}] starting…\n`);
  issAgentEs = new EventSource(url);
  issAgentEs.onmessage = (e)=>{
    const ev = JSON.parse(e.data);
    if(ev.type === "text_delta") issAgentLogAppend(ev.text||"");
    else if(ev.type === "tool_use") issAgentLogAppend(`\n[tool: ${ev.name}]\n`);
    else if(ev.type === "tool_result") issAgentLogAppend(`→ ${(ev.output||"").slice(0,300)}\n`);
    else if(ev.type === "error") issAgentLogAppend(`\n[error] ${ev.message||""}\n`);
    else if(ev.type === "done") issAgentStop(ev.reason||"done");
    else if(ev.type === "status") issAgentLogAppend(`[${ev.label||"agent"}] ${ev.state}\n`);
    else if(ev.type === "usage") issAgentLogAppend(`\n[usage] in=${ev.input_tokens} out=${ev.output_tokens}\n`);
  };
  issAgentEs.onerror = ()=>{ issAgentLogAppend("\n[stream error]\n"); issAgentStop("error"); };
}

async function loadDocuments(){
  _showDocAllView();
  const r=await fetch("/api/documents"); const {documents}=await r.json();
  const list=$("d-list"); list.innerHTML="";
  if(!documents.length){ list.innerHTML='
    No reports yet.
    '; return; }
  // Build tree: group by folder
  const rootItems=documents.filter(d=>!d.folder);
  const folders={};
  for(const d of documents){
    if(d.folder){
      if(!folders[d.folder]) folders[d.folder]=[];
      folders[d.folder].push(d);
    }
  }
  function addDocItem(doc, parent){
    const it=el("div","qitem");
    it._docPath = doc.path||doc.name;
    it.innerHTML=`${esc(doc.name)}${esc(doc.title)} · ${doc.modified}`;
    it.onclick=()=>selectDocument(doc.path||doc.name, it);
    parent.appendChild(it);
  }
  // Root-level files first
  for(const d of rootItems) addDocItem(d, list);
  // Folders (collapsible)
  for(const [folderName, items] of Object.entries(folders).sort()){
    const fold=el("div","doc-folder");
    fold.innerHTML=`▾ 📁 ${esc(folderName)} (${items.length})`;
    const children=el("div","doc-folder-children");
    for(const d of items) addDocItem(d, children);
    fold.onclick=e=>{
      if(e.target!==fold && e.target.closest(".qitem")) return; // don't collapse when clicking item
      fold.classList.toggle("collapsed");
      children.style.display=fold.classList.contains("collapsed")?"none":"";
    };
    list.appendChild(fold);
    list.appendChild(children);
  }
}

function _heroRenderDiagram(sections, budgetTok, usedTok){
  const diag=$("hero-diagram");
  if(!diag) return;
  diag.innerHTML="";

  // The diagram height corresponds to budgetTok.
  // Each enabled section gets a colored block proportional to its tokens.
  const diagH = diag.clientHeight || 400;

  ceiling.title="Context budget ceiling";
  diag.appendChild(ceiling);

  // Sort by order for visual stacking
  const sorted=[...sections].sort((a,b)=>a.order-b.order);

    const tok=s.tokens||0;
    if(!tok && !s.enabled) continue;

    block.title=`${s.name}: ${_heroTokFmt(tok)} tokens`;

    // Label inside block
    const lbl=document.createElement("div");
    lbl.style.cssText="position:absolute;inset:0;display:flex;align-items:center;padding:0 6px;font-size:9px;font-weight:700;color:#fff;text-shadow:0 1px 2px rgba(0,0,0,.5);white-space:nowrap;overflow:hidden;text-overflow:ellipsis;pointer-events:none";
    lbl.textContent=`${s.name} · ${_heroTokFmt(tok)}`;
    block.appendChild(lbl);

    if(s.truncated){
      const badge=document.createElement("div");
      badge.style.cssText="position:absolute;top:2px;right:4px;font-size:8px;background:rgba(0,0,0,.5);color:#ffa;padding:1px 4px;border-radius:3px;font-weight:700";
      badge.textContent="TRUNCATED";
      block.appendChild(badge);
    }

    block.onclick=()=>heroSelectSection(s.id);
    diag.appendChild(block);
  }

  // Unused budget space
  const unused=Math.max(0,budgetTok-usedTok);
  if(unused>0){
    const gap=document.createElement("div");
    const heightPct=unused/budgetTok*100;
    gap.style.cssText=`height:${heightPct.toFixed(2)}%;min-height:4px;border:1px dashed rgba(255,255,255,.15);border-radius:3px;flex-shrink:0`;

  }
}

async function heroRun(){
  if(!selected){ alert("Select a question first."); return; }
  const budget=parseInt($("hero-budget-sel")?.value||"180000");
  const enabledIds=_heroSections.filter(s=>s.enabled).map(s=>s.id);
  if(!enabledIds.length){ alert("No sections enabled."); return; }

  // Switch to output view
  $("hero-preview-hint") && ($("hero-preview-hint").style.display="none");
  $("hero-preview") && ($("hero-preview").style.display="none");
  const outWrap=$("hero-output-wrap");
  if(outWrap){ outWrap.style.display="flex"; }
  const out=$("hero-output"); if(out){ out.textContent=""; }
  $("hero-run-btn") && ($("hero-run-btn").style.display="none");
  $("hero-stop-btn") && ($("hero-stop-btn").style.display="");

  const ds=selectedDataset||"first_proof_1";
  const statsEl=$("hero-output-stats"); if(statsEl) statsEl.textContent="streaming…";

  _heroEvtSource=null;
  // Use SSE via fetch + ReadableStream for POST
  let fullText="";
  try{
    const resp=await fetch(`/api/hero/run/${selected}?dataset=${encodeURIComponent(ds)}`,{
      method:"POST",
      headers:{"content-type":"application/json"},
      body:JSON.stringify({enabled_sections:enabledIds,budget}),
    });
    const reader=resp.body.getReader();
    const decoder=new TextDecoder();
    let buf="";
    while(true){
      const {done,value}=await reader.read();
      if(done) break;
      buf+=decoder.decode(value,{stream:true});
      const lines=buf.split("
");
      buf=lines.pop();
      for(const line of lines){
        if(!line.startsWith("data: ")) continue;
        try{
          const ev=JSON.parse(line.slice(6));
          if(ev.type==="text"){ fullText+=ev.text; if(out) out.textContent=fullText; out?.scrollTo(0,out.scrollHeight); }
          else if(ev.type==="hero_done"){
            if(statsEl) statsEl.textContent=`${_heroTokFmt(ev.input_tokens||0)} in · ${_heroTokFmt(ev.output_tokens||0)} out`;
          }
          else if(ev.type==="error"){
            if(out) out.textContent+="\n\n[ERROR] "+ev.error;
          }
        }catch(e){}
      }
    }
  }catch(e){ if(out) out.textContent+="\n\n[STREAM ERROR] "+e; }

  $("hero-run-btn") && ($("hero-run-btn").style.display="");
  $("hero-stop-btn") && ($("hero-stop-btn").style.display="none");
}

// ---- SSE runner for meet agents ----
function meetRunSse(endpoint, label){
  if(meetAgentEs){ meetAgentEs.close(); meetAgentEs=null; }
  const bar=$("meet-agent-bar"), logEl=$("meet-agent-log");
  bar.style.display="flex"; $("meet-agent-label").textContent=label;
  $("meet-agent-status").textContent=""; logEl.textContent="";
  const es=new EventSource(endpoint); meetAgentEs=es;
  es.onmessage=async(ev)=>{
    try{
      const d=JSON.parse(ev.data);
      if(d.type==="text_delta"||d.type==="thinking_delta") logEl.textContent+=d.data.text||"";
      if(d.type==="status") $("meet-agent-status").textContent=d.data.label||"";
      if(d.type==="error") logEl.textContent+="\n[ERROR] "+(d.data.message||"");
      if(d.type==="done"){
        es.close(); meetAgentEs=null; bar.style.display="none";
        await refreshMeetRoom();
        await loadMeetRooms(selected);
        if(_meetAutoQueue.length){ const next=_meetAutoQueue.shift(); setTimeout(()=>next(),400); }
      }
    }catch(e){}
  };
  es.onerror=()=>{ es.close(); meetAgentEs=null; bar.style.display="none"; };
}

function _insShowPanel(level){

function _renderInsightCard(data, container, emptyEl){
  if(!data || data.error){
    container.style.display = "none";
    emptyEl.style.display = "";
    return;
  }
  emptyEl.style.display = "none";
  container.style.display = "";

  const genAt = data.generated_at
    ? new Date(data.generated_at).toLocaleString(undefined,{dateStyle:"medium",timeStyle:"short"})
    : "unknown";
  const levelLabel = data.level==="system" ? "System"
    : data.level==="dataset" ? `Dataset: ${esc(data.dataset||data.id||"")}`
    : `Question: ${esc(data.id||"")} (${esc(data.dataset||"")})`;

  const problems = (data.problems||[]);
  const highlights = (data.highlights||[]);
  const todos = (data.suggested_todos||[]);

  const problemsHtml = problems.length
    ? problems.map(p=>`
    ⚠${esc(p)}
    `).join("")
    : `
    ✓No problems identified.
    `;

  const highlightsHtml = highlights.length
    ? highlights.map(h=>`
    ✨${esc(h)}
    `).join("")
    : `
    —None yet.
    `;

  const todosHtml = todos.length
    ? todos.map(t=>{
        const p=(t.priority||"medium").toLowerCase();
        return `
    ${p.toUpperCase()}${esc(t.title||t)}
    `;
      }).join("")
    : `
    —None suggested.
    `;

  container.innerHTML = `
    
    
      
    ${levelLabel}
    
      ${data.summary ? `
    ${esc(data.summary)}
    ` : ""}
      
    
        
    ⚠ Problems
    
        ${problemsHtml}
      
    
      
    
        
    ✨ Highlights
    
        ${highlightsHtml}
      
    
      
    
        
    ✅ Suggested TODOs
    
        ${todosHtml}
      
    
      
    Generated by document-manager · ${genAt}
    
    
    `;
}

    }
