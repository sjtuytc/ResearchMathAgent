\title{The Hilbert series of the successive subalgebras
$\Q\langle e_1,\dots ,e_m\rangle$ of Grassmannian cohomology}
\maketitle

Throughout, $\Lambda=\Lambda_{\Q}$ is the ring of symmetric functions over $\Q$,
$s_\lambda$ the Schur functions, $e_i$ the elementary and $h_i$ the complete
homogeneous symmetric functions, all graded by weight $|\lambda|=\sum_i\lambda_i$.
For a partition $\lambda$ we write $\ell(\lambda)$ for its number of (positive)
parts, $\lambda_1$ for its largest part, $\lambda'$ for the conjugate partition
(so $\lambda'_1=\ell(\lambda)$ is the longest column height), and
$\lambda\subseteq k\times\ell$ when
$\ell(\lambda)\le k$ and $\lambda_1\le\ell$. We use the standard $q$-integers
$[n]_q=\frac{1-q^n}{1-q}$, $[n]_q!=\prod_{i=1}^n[i]_q$, and Gaussian binomial
coefficients $\qbin{n}{k}=\frac{[n]_q!}{[k]_q![n-k]_q!}$.

\medskip
\noindent\textbf{The ring.} As recalled in the statement,
\[
\rkl=H^{*}(\Gr(k,\mathbb{C}^{k+\ell});\Q)\cong
\Lambda\big/ I^{k,\ell},
\qquad
I^{k,\ell}=\big\langle\, s_\lambda \ :\ \lambda\not\subseteq k\times\ell\,\big\rangle ,
\]
and a $\Q$-basis of $\rkl$ is $\{\,\overline{s_\lambda}\ :\ \lambda\subseteq k\times\ell\,\}$;
hence
\begin{equation}\label{eq:HilbR}
\Hilb(\rkl;q)=\sum_{\lambda\subseteq k\times\ell}q^{|\lambda|}=\qbin{k+\ell}{k}.
\end{equation}
For $0\le m\le\ell$ we set $\Rklm=\Q\langle e_1,\dots,e_m\rangle\subseteq\rkl$
(with $R^{k,\ell,0}=\Q$), a graded subalgebra, and
\[
Q_{k,\ell,m}(q):=\Hilb\!\left(\Rklm/R^{k,\ell,m-1};q\right)
=\Hilb(\Rklm;q)-\Hilb(R^{k,\ell,m-1};q),
\]
where the last equality holds because $R^{k,\ell,m-1}\subseteq\Rklm$ are graded
subspaces, so each graded piece of the quotient has dimension the difference of
dimensions. We must prove, for all $k\ge1$ and $1\le m\le\ell$,
\begin{equation}\label{eq:main}
Q_{k,\ell,m}(q)=q^{m}\,\qbin{\ell}{m}\,
\sum_{j=0}^{k-m} q^{\,j(\ell-m+1)}\,\qbin{m+j-1}{j}
=:F_{k,\ell,m}(q).
\end{equation}

\bigskip
The proof proceeds by induction on $k$ (for fixed $\ell$ and $m$). The structural
input is a surjection $\pi\colon\rkl\twoheadrightarrow R^{k-1,\ell}$ which restricts
to the subalgebras and whose kernel is principal. We first record elementary
reductions and the behaviour of the right-hand side, then develop the structural
lemmas, and finally combine them.

\section{Elementary reductions and the right-hand side}

\begin{lemma}\label{lem:trivial}
If $m\ge k$ then $\Rklm=\rkl$. In particular $R^{k,\ell,\ell}=\rkl$, and the
nontrivial range of \eqref{eq:main} is $1\le m\le\min(k,\ell)$; the cases
$\ell\ge m>k$ do not occur because $m\le\ell$ and we take $k\ge m$ in the
induction (cases $k<m$ being subsumed by $\Rklm=\rkl$).
\end{lemma}

\begin{proof}
The ring $\rkl$ is generated by $e_1,\dots,e_k$. If $m\ge k$ then
$e_1,\dots,e_m$ already contain all generators, so $\Rklm=\rkl$.
For $R^{k,\ell,\ell}=\rkl$ when $k>\ell$: the generating-function identity
$\big(\sum_{i\ge0}(-1)^i e_i z^i\big)\big(\sum_{j\ge0}h_j z^j\big)=1$ gives, for
$n\ge1$,
\[
\sum_{i=0}^{n}(-1)^i e_i\,h_{n-i}=0 .
\]
For $1\le j\le\ell$, induction on $j$ via this relation (with $n=j$) expresses
$h_j$ as a polynomial in $e_1,\dots,e_j$. For $n=\ell+1,\dots,k$, using
$h_{\ell+1}=\dots=h_{\ell+k}=0$ in $\rkl$ and that $h_1,\dots,h_\ell$ are
polynomials in $e_1,\dots,e_\ell$, the relation with this $n$ solves $(-1)^n e_n$
(its top term, since $h_0=1$) in terms of $e_0,\dots,e_{n-1}$ and $h_1,\dots,h_\ell$,
hence in terms of $e_1,\dots,e_{n-1}$. By induction on $n$ each
$e_{\ell+1},\dots,e_k$ lies in $\Q\langle e_1,\dots,e_\ell\rangle$, so
$R^{k,\ell,\ell}=\rkl$.
\end{proof}

\begin{lemma}[Recursion for the right-hand side]\label{lem:RHSrec}
For all $1\le m\le\ell$ and $k>m$,
\begin{equation}\label{eq:RHSrec}
F_{k,\ell,m}(q)-F_{k-1,\ell,m}(q)
= q^{\,m+(k-m)(\ell-m+1)}\,\qbin{\ell}{m}\,\qbin{k-1}{m-1},
\end{equation}
and the base value is
\begin{equation}\label{eq:RHSbase}
F_{m,\ell,m}(q)=q^{m}\,\qbin{\ell}{m}.
\end{equation}
\end{lemma}

\begin{proof}
In $F_{k,\ell,m}$ the summand $\qbin{m+j-1}{j}$ does not depend on $k$, and the
$j$-range is $0\le j\le k-m$. Increasing $k$ by one only appends the single term
$j=k-m$:
\[
F_{k,\ell,m}-F_{k-1,\ell,m}
= q^{m}\qbin{\ell}{m}\,q^{(k-m)(\ell-m+1)}\qbin{m+(k-m)-1}{\,k-m\,}.
\]
Now $\qbin{k-1}{k-m}=\qbin{k-1}{m-1}$ by the symmetry $\qbin{n}{a}=\qbin{n}{n-a}$,
which gives \eqref{eq:RHSrec}. For \eqref{eq:RHSbase}, when $k=m$ the $j$-range is
the single value $j=0$, with $q^{0}\qbin{m-1}{0}=1$, so
$F_{m,\ell,m}=q^{m}\qbin{\ell}{m}$.
\end{proof}

Thus it suffices to prove that the left-hand side $Q_{k,\ell,m}$ obeys the same
recursion and base value:
\begin{align}
\label{eq:LHSrec}
Q_{k,\ell,m}(q)-Q_{k-1,\ell,m}(q)
&= q^{\,m+(k-m)(\ell-m+1)}\,\qbin{\ell}{m}\,\qbin{k-1}{m-1},
&&(k>m),\\
\label{eq:LHSbase}
Q_{m,\ell,m}(q)&=q^{m}\,\qbin{\ell}{m}.
\end{align}

\begin{remark}[Verification]\label{rem:verif}
Both \eqref{eq:main}, and the equivalent pair \eqref{eq:LHSrec}--\eqref{eq:LHSbase},
were checked symbolically for all $1\le k,\ell\le5$ and $1\le m\le\ell$ by computing
$\Rklm$ inside the Schur basis of $\rkl$ (multiplication by $e_i$ is the Pieri rule
for vertical strips, truncated to $k\times\ell$) and row-reducing over $\Q$; all
cases agree. These computations also confirm the structural objects used below
(the surjection $\pi$, its kernel $e_k\rkl$, and the base-case dimension count).
\end{remark}

\section{The column-height span $F_m$ is \emph{not} the subalgebra}
\label{sec:Fm}

The most natural attempt to compute $Q_{k,\ell,m}$ would be to exhibit a
Schur-class basis of $\Rklm$ cut out by a column-height bound. Define the
\emph{column-height span}
\[
F_m \;:=\; \operatorname{span}_{\Q}\bigl\{\,\overline{s_\lambda}\ :\
\lambda\subseteq k\times\ell,\ \lambda'_1\le m\,\bigr\}\subseteq\rkl ,
\]
the span of those Schur classes whose Young diagram has no column taller than $m$
(recall $\lambda'_1=\ell(\lambda)$). One might hope that $F_m=\Rklm$, which would
give the clean basis $\{\overline{s_\lambda}:\lambda\subseteq k\times\ell,\
\lambda'_1=m\}$ of $\Rklm/R^{k,\ell,m-1}$ with Hilbert series
$\sum_{\lambda'_1=m}q^{|\lambda|}$. The purpose of this section is to prove rigorously
that \emph{this hope is false}: \emph{both} inclusions $\Rklm\subseteq F_m$ and
$F_m\subseteq\Rklm$ fail, so $\Rklm$ has no Schur-class basis selected by a
column-height threshold, and the route to \eqref{eq:main} must avoid any such basis.
This isolates and refutes the load-bearing column-height lemma proposed for this part,
so that no later argument relies on it.

\begin{lemma}[The inclusion $\Rklm\subseteq F_m$ fails]\label{lem:RnotinF}
The vertical-strip Pieri rule reads, for a partition $\mu$ with $\ell(\mu)\le k$,
\begin{equation}\label{eq:pierie}
e_i\cdot s_\mu=\sum_{\nu}s_\nu ,
\end{equation}
the sum over $\nu\supseteq\mu$ with $\nu/\mu$ a vertical $i$-strip (at most one new
box in each row). Consequently:
\begin{enumerate}
\item[\textup{(i)}] At $(k,\ell,m)=(2,1,1)$: $e_1^2=s_{(2)}+s_{(1,1)}$ in $\Lambda$,
and in $R^{2,1}$ one has $\overline{s_{(2)}}=0$ (as $2>\ell=1$), so
$\overline{e_1^{\,2}}=\overline{s_{(1,1)}}\neq0$ lies in $R^{2,1,1}$; but
$(1,1)'_1=2>1$, so $\overline{s_{(1,1)}}\notin F_1$. Hence $R^{2,1,1}\not\subseteq F_1$.
\item[\textup{(ii)}] More generally, whenever $2m\le k$ \textup{(}and $\ell\ge m$\textup{)},
$\overline{e_m^{\,2}}\in\Rklm$ but $\overline{e_m^{\,2}}\notin F_m$.
\end{enumerate}
In particular $\Rklm\not\subseteq F_m$ for all $(k,\ell,m)$ with $2m\le k$.
\end{lemma}

\begin{proof}
Identity \eqref{eq:pierie} is the classical Pieri rule for $e_i=s_{(1^i)}$
(vertical strips); see e.g.\ Macdonald, \emph{Symmetric Functions and Hall
Polynomials}, I.5. Part (i) is the direct computation just displayed.

For (ii), apply \eqref{eq:pierie} with $\mu=(1^m)$: the partitions $\nu$ with
$\nu/(1^m)$ a vertical $m$-strip are exactly those obtained by adding at most one box
to each of the $m$ rows (and possibly opening a new row of length $1$ below); the
particular choice of stacking a second full column of height $m$ underneath gives
$\nu=(1^{2m})$, which is a vertical $m$-strip over $(1^m)$. Each such $\nu$ occurs
with coefficient $1$, and $(1^{2m})$ is the unique term of $e_m^2$ of shape having
$2m$ rows of length $1$; hence the coefficient of $s_{(1^{2m})}$ in $e_m^2$ equals
$1$. If $2m\le k$ then $(1^{2m})\subseteq k\times\ell$ (using $\ell\ge m\ge1$ for the
part-size bound), so $\overline{s_{(1^{2m})}}$ is a nonzero element of the Schur basis
of $\rkl$, distinct from every other shape appearing in $\overline{e_m^{\,2}}$. Thus
the Schur expansion of $\overline{e_m^{\,2}}\in\Rklm$ has nonzero
$\overline{s_{(1^{2m})}}$-coefficient, and since $(1^{2m})'_1=2m>m$, this element is
not a $\Q$-combination of Schur classes of column height $\le m$, i.e.\
$\overline{e_m^{\,2}}\notin F_m$.

(We confirmed (ii) directly: in $R^{4,2}$, $\overline{e_2^{\,2}}=
\overline{s_{(2,2)}}+\overline{s_{(2,1,1)}}+\overline{s_{(1,1,1,1)}}$, whose
$s_{(1^4)}$-coefficient is $1$.)
\end{proof}

\begin{lemma}[The inclusion $F_m\subseteq\Rklm$ fails]\label{lem:FnotinR}
$F_m\not\subseteq\Rklm$ in general; e.g.\ for $(k,\ell,m)=(2,2,1)$ one has
$\overline{s_{(2)}}\in F_1$ but $\overline{s_{(2)}}\notin R^{2,2,1}=\Q\langle e_1\rangle$.
\end{lemma}

\begin{proof}
The shape $(2)$ has longest column $(2)'_1=1\le1$, so $\overline{s_{(2)}}\in F_1$.
On the other hand $s_{(2)}=e_1^{\,2}-e_2$ in $\Lambda$, and the degree-$2$ part of
$R^{2,2,1}=\Q\langle e_1\rangle$ is the line $\Q\cdot\overline{e_1^{\,2}}$. In the
Schur basis of $R^{2,2}$ (with distinct basis vectors $\overline{s_{(2)}}$ and
$\overline{s_{(1,1)}}$), the Pieri rule gives
$\overline{e_1^{\,2}}=\overline{s_{(2)}}+\overline{s_{(1,1)}}$, which is not a scalar
multiple of $\overline{s_{(2)}}$. Hence $\overline{s_{(2)}}\notin
\Q\cdot\overline{e_1^{\,2}}=(R^{2,2,1})_2$, so $\overline{s_{(2)}}\notin R^{2,2,1}$
and $F_1\not\subseteq R^{2,2,1}$.

More conceptually, the dual Jacobi--Trudi identity
$s_\lambda=\det\bigl(e_{\lambda'_i-i+j}\bigr)_{1\le i,j\le\lambda_1}$ expresses
$s_\lambda$ as a polynomial in $e_1,\dots,e_{N(\lambda)}$ with
$N(\lambda)=\lambda'_1+\lambda_1-1$ (the largest index $\lambda'_i-i+j$, attained at
$i=1,\ j=\lambda_1$). The generators needed to realize $s_\lambda$ are thus controlled
by \emph{both} the column height and the row width, not by the column height alone, so
column height $\le m$ does not force membership in $\Q\langle e_1,\dots,e_m\rangle$.
\end{proof}

\begin{proposition}[No column-height Schur basis exists]\label{prop:nocolbasis}
For $1\le m\le\ell$, the subspaces $F_m$ and $\Rklm$ are in general incomparable
(Lemmas~\ref{lem:RnotinF} and~\ref{lem:FnotinR}). Consequently
$\{\overline{s_\lambda}:\lambda'_1=m\}$ is neither a subset of, nor a spanning set
for, $\Rklm/R^{k,\ell,m-1}$, and the naive series $\sum_{\lambda'_1=m}q^{|\lambda|}$
does \emph{not} equal $Q_{k,\ell,m}(q)$ in general. Moreover no monotone shape
statistic detects membership in $\Rklm$.
\end{proposition}

\begin{proof}
Incomparability is Lemmas~\ref{lem:RnotinF}--\ref{lem:FnotinR}. Numerically, for
$(k,\ell,m)=(3,3,2)$ one has $\dim_\Q\Rklm=19$ while
$\dim_\Q F_2=\#\{\lambda\subseteq3\times3:\ \ell(\lambda)\le2\}=10$ (and
$\dim_\Q R^{3,3}=20$); in particular the would-be quotient basis
$\{\overline{s_\lambda}:\lambda'_1=2\}$ has the wrong cardinality. Concretely the two
generating functions differ already here:
\[
\sum_{\ell(\lambda)=2}q^{|\lambda|}=q^2+q^3+2q^4+q^5+q^6,
\qquad
Q_{3,3,2}(q)=q^2+q^3+2q^4+2q^5+2q^6+q^7 ,
\]
the latter agreeing with $F_{3,3,2}(q)$ of \eqref{eq:main}. Finally, the minimal
$m$ with $\overline{s_\lambda}\in\Rklm$ is not a function of $\lambda_1$, of
$\ell(\lambda)$, or of $|\lambda|$ alone: in $R^{4,4}$ one has
$\overline{s_{(1,1)}}\in\Rklm$ only for $m\ge2$, whereas
$\overline{s_{(4,4,4,3)}}\in\Rklm$ already for $m\ge1$. Hence $\Rklm$ admits no
Schur-class basis cut out by a column-height (or row-width, or size) threshold.
All stated dimensions and series were confirmed by direct computation in the Schur
basis of $\rkl$ (closing $\{1\}$ under multiplication by $e_1,\dots,e_m$ via the
truncated vertical-strip Pieri rule and row-reducing over $\Q$).
\end{proof}

\begin{remark}[What replaces the column-height basis]\label{rem:replacement}
Although no column-height (indeed no monotone shape-statistic) condition cuts out a
Schur-class basis of $\Rklm$, the \emph{count} asserted in \eqref{eq:main} is correct.
The correct route, developed in the next section, is the comparison surjection
$\pi\colon\rkl\twoheadrightarrow R^{k-1,\ell}$, which yields the recursion
\eqref{eq:LHSrec} for $Q_{k,\ell,m}$ \emph{without} producing any explicit Schur or
monomial basis of the subalgebra. We have verified
$\Hilb(\Rklm/R^{k,\ell,m-1};q)=F_{k,\ell,m}(q)$ for all $1\le k,\ell\le5$,
$1\le m\le\ell$ (Remark~\ref{rem:verif}); this is the content the column-height
lemma was meant to supply, now decoupled from the (false) basis claim.
\end{remark}

\section{The comparison surjection and its kernel}

\begin{lemma}[Comparison surjection]\label{lem:pi}
For $k\ge2$ there is a surjective graded $\Q$-algebra homomorphism
\[
\pi\colon \rkl\twoheadrightarrow R^{k-1,\ell},\qquad \pi(e_i)=e_i\ (1\le i\le k-1),
\quad \pi(e_k)=0,
\]
induced on quotients by $\mathrm{id}_\Lambda$, using $I^{k,\ell}\subseteq I^{k-1,\ell}$.
Its kernel is the principal ideal
\begin{equation}\label{eq:kerpi}
\ker\pi=e_k\cdot\rkl
=\operatorname{span}_{\Q}\{\,\overline{s_\lambda}\ :\ \lambda\subseteq k\times\ell,\ \ell(\lambda)=k\,\},
\end{equation}
and as a graded vector space
\begin{equation}\label{eq:kerHilb}
\Hilb(\ker\pi;q)=q^{k}\,\qbin{k-1+\ell}{k}.
\end{equation}
\end{lemma}

\begin{proof}
Since $(k-1)\times\ell\subseteq k\times\ell$, every $\lambda\not\subseteq k\times\ell$
also satisfies $\lambda\not\subseteq(k-1)\times\ell$, so $I^{k,\ell}\subseteq
I^{k-1,\ell}$ as ideals of $\Lambda$. Hence the identity of $\Lambda$ descends to a
surjective graded algebra map $\pi\colon\Lambda/I^{k,\ell}\to\Lambda/I^{k-1,\ell}$,
i.e.\ $\pi\colon\rkl\twoheadrightarrow R^{k-1,\ell}$. As $e_i$ maps to $e_i$ in
$\Lambda$, we get $\pi(\overline{e_i})=\overline{e_i}$; and
$\overline{e_k}=\overline{s_{(1^k)}}=0$ in $R^{k-1,\ell}$ because $(1^k)$ has $k>k-1$
rows, i.e.\ $(1^k)\not\subseteq(k-1)\times\ell$.

The kernel of $\pi$ is the image in $\rkl$ of $I^{k-1,\ell}$, which (modulo
$I^{k,\ell}$) is spanned by those $\overline{s_\lambda}$ with $\lambda\subseteq
k\times\ell$ but $\lambda\not\subseteq(k-1)\times\ell$, i.e.\ with exactly $k$ rows
($\ell(\lambda)=k$). This proves the second equality in \eqref{eq:kerpi}.

For the first equality, $e_k=s_{(1^k)}$, and the vertical-strip Pieri rule gives in
$\Lambda$
\[
e_k\cdot s_\mu=s_{\mu+(1^k)}\qquad(\ell(\mu)\le k),
\]
the only vertical strip of size $k$ addable to a partition of length $\le k$ being a
full extra first column. Reducing modulo $I^{k,\ell}$: if $\mu\subseteq k\times\ell$
and $\mu_1\le\ell-1$ then $\mu+(1^k)\subseteq k\times\ell$ and has exactly $k$ rows;
if $\mu_1=\ell$ then $\mu+(1^k)$ has first part $\ell+1>\ell$, so
$\overline{s_{\mu+(1^k)}}=0$. Therefore
\[
e_k\cdot\rkl=\operatorname{span}_{\Q}
\{\,\overline{s_{\mu+(1^k)}}\ :\ \mu\subseteq k\times(\ell-1)\,\}
=\operatorname{span}_{\Q}\{\,\overline{s_\lambda}\ :\ \lambda\subseteq k\times\ell,\ \ell(\lambda)=k\,\},
\]
the last equality by the bijection $\lambda\mapsto\mu=\lambda-(1^k)$ between
$k$-row partitions in $k\times\ell$ and partitions in $k\times(\ell-1)$. This is
exactly $\ker\pi$, proving \eqref{eq:kerpi}. Finally, since $\overline{s_{\mu}}\mapsto
\overline{s_{\mu+(1^k)}}$ raises degree by $k$ and is a bijection onto a basis of
$\ker\pi$, \eqref{eq:kerHilb} follows from $\Hilb(R^{k,\ell-1};q)=\qbin{k-1+\ell}{k}$
(equation \eqref{eq:HilbR} with $\ell$ replaced by $\ell-1$).
\end{proof}

\begin{corollary}[$\pi$ respects the filtration]\label{cor:pifiltr}
For $0\le m\le \ell$ and $k\ge2$ we have $\pi(\Rklm)=R^{k-1,\ell,m}$. Consequently
$\pi$ induces a surjection of graded vector spaces
\[
\overline{\pi}\colon \Rklm/R^{k,\ell,m-1}\twoheadrightarrow
R^{k-1,\ell,m}/R^{k-1,\ell,m-1},
\]
and therefore
\begin{equation}\label{eq:Qdiff}
Q_{k,\ell,m}(q)-Q_{k-1,\ell,m}(q)=\Hilb\big(\ker\overline{\pi};q\big),
\qquad
\ker\overline{\pi}=\frac{\pi^{-1}(R^{k-1,\ell,m-1})\cap \Rklm}{R^{k,\ell,m-1}} .
\end{equation}
\end{corollary}

\begin{proof}
For the nontrivial step we have $m\le k-1$, so $e_1,\dots,e_m$ are among the
generators $e_1,\dots,e_{k-1}$ that $\pi$ sends to the corresponding generators of
$R^{k-1,\ell}$. As $\pi$ is an algebra homomorphism,
$\pi\big(\Q\langle e_1,\dots,e_m\rangle\big)=\Q\langle e_1,\dots,e_m\rangle$ inside
$R^{k-1,\ell}$, i.e.\ $\pi(\Rklm)=R^{k-1,\ell,m}$, and likewise
$\pi(R^{k,\ell,m-1})=R^{k-1,\ell,m-1}$. Thus $\pi$ carries the pair
$(R^{k,\ell,m-1}\subseteq\Rklm)$ onto $(R^{k-1,\ell,m-1}\subseteq R^{k-1,\ell,m})$
and induces a surjection $\overline{\pi}$ on the quotients. For a surjective map of
graded vector spaces, $\Hilb(\text{domain})=\Hilb(\text{image})+\Hilb(\ker)
=\Hilb(\text{codomain})+\Hilb(\ker)$, which is the first identity in \eqref{eq:Qdiff};
the description of $\ker\overline\pi$ is immediate from its definition.
\end{proof}

By Corollary~\ref{cor:pifiltr}, the inductive step \eqref{eq:LHSrec} is equivalent
to the following Hilbert-series identity for the kernel of $\overline\pi$.

\begin{proposition}[Kernel of $\overline\pi$]\label{prop:kernel}
For $1\le m\le\ell$ and $m<k$,
\begin{equation}\label{eq:kernelHilb}
\Hilb\big(\ker\overline\pi;q\big)
= q^{\,m+(k-m)(\ell-m+1)}\,\qbin{\ell}{m}\,\qbin{k-1}{m-1}.
\end{equation}
\end{proposition}

\section{Reduction to two Hilbert-series identities}

The two statements that complete the proof are Proposition~\ref{prop:kernel}
(equivalent to the inductive step \eqref{eq:LHSrec}) and the base case
\eqref{eq:LHSbase}. We record their precise content and the structural reductions
established.

\subsection*{Reformulation of Proposition~\ref{prop:kernel}}
By \eqref{eq:Qdiff} and $\ker\pi=e_k\rkl$ (Lemma~\ref{lem:pi}),
\[
\ker\overline\pi\ \cong\
\frac{\{x\in\Rklm:\ \pi(x)\in R^{k-1,\ell,m-1}\}}{R^{k,\ell,m-1}} .
\]
The target series in \eqref{eq:kernelHilb} factors as
$q^{\,m+(k-m)(\ell-m+1)}\cdot\qbin{\ell}{m}\cdot\qbin{k-1}{m-1}$, where
$\qbin{\ell}{m}=\sum_{\nu\subseteq m\times(\ell-m)}q^{|\nu|}$ is the Hilbert series
of partitions in an $m\times(\ell-m)$ box and
$\qbin{k-1}{m-1}=\sum_{\rho\subseteq(m-1)\times(k-m)}q^{|\rho|}$ that of partitions
in an $(m-1)\times(k-m)$ box, the whole shifted in degree by $m+(k-m)(\ell-m+1)$.

\subsection*{Reformulation of the base case \eqref{eq:LHSbase}}
For $k=m$, $\Rklm=R^{m,\ell,m}=R^{m,\ell}$ (Lemma~\ref{lem:trivial}), so
\[
Q_{m,\ell,m}(q)=\qbin{m+\ell}{m}-\Hilb(R^{m,\ell,m-1};q),
\]
and \eqref{eq:LHSbase} is equivalent to
\begin{equation}\label{eq:basecum}
\Hilb(R^{m,\ell,m-1};q)=\qbin{m+\ell}{m}-q^{m}\qbin{\ell}{m}.
\end{equation}
Here $q^{m}\qbin{\ell}{m}$ is the generating function of partitions with exactly $m$
parts, each in $\{1,\dots,\ell-m+1\}$; equivalently, \eqref{eq:LHSbase} asserts that
$\dim_\Q\big(R^{m,\ell}/R^{m,\ell,m-1}\big)_d$ equals the number of partitions of $d$
with exactly $m$ parts, all $\le\ell-m+1$.

\subsection*{Assembling the induction}
Granting Proposition~\ref{prop:kernel} and the base case \eqref{eq:LHSbase}, the main
identity \eqref{eq:main} follows. Fix $\ell$ and $1\le m\le\ell$ and induct on
$k\ge m$. For $k=m$, \eqref{eq:LHSbase} and \eqref{eq:RHSbase} give
$Q_{m,\ell,m}=q^m\qbin{\ell}{m}=F_{m,\ell,m}$. For $k>m$,
Corollary~\ref{cor:pifiltr} and Proposition~\ref{prop:kernel} give
\[
Q_{k,\ell,m}=Q_{k-1,\ell,m}+q^{\,m+(k-m)(\ell-m+1)}\qbin{\ell}{m}\qbin{k-1}{m-1}
=F_{k-1,\ell,m}+\big(F_{k,\ell,m}-F_{k-1,\ell,m}\big)=F_{k,\ell,m},
\]
using the inductive hypothesis $Q_{k-1,\ell,m}=F_{k-1,\ell,m}$ and
Lemma~\ref{lem:RHSrec}. This completes the induction, hence \eqref{eq:main}, for all
$k\ge m$; for $k<m$ the claim is the degenerate case where $\Rklm=\rkl$ and
\eqref{eq:main} reduces to $m=\ell=k$ being excluded, so the stated range is covered.

\section{Boundary cases and the global telescope}
\label{sec:boundary}

We record three consistency checks demanded for completeness: the degenerate
range $m>k$, the smallest nontrivial value $m=1$, and the global telescope. The
first two are immediate from the closed form $F_{k,\ell,m}$; the telescope is a
nontrivial $q$-identity which we both derive as a corollary of the main theorem
\eqref{eq:main} \emph{and} prove independently from the right-hand side, via the
$q$-Vandermonde convolution. Throughout we use the convention that
$\qbin{n}{a}=0$ unless $0\le a\le n$, and that an empty sum is $0$.

\begin{lemma}[Degenerate range $m>k$]\label{lem:mbigk}
For $m>k$ the right-hand side of \eqref{eq:main} is the empty sum, hence
$F_{k,\ell,m}=0$; and the left-hand side $Q_{k,\ell,m}=0$ as well, consistently.
\end{lemma}

\begin{proof}
In $F_{k,\ell,m}=q^{m}\qbin{\ell}{m}\sum_{j=0}^{k-m}q^{j(\ell-m+1)}\qbin{m+j-1}{j}$
the summation index $j$ runs over $0\le j\le k-m$; when $m>k$ this range is empty,
so the inner sum is $0$ and $F_{k,\ell,m}=0$.

On the algebra side, $\rkl$ is generated by $e_1,\dots,e_k$, and for $k<i\le\ell$
the class $e_i=\overline{s_{(1^i)}}=0$ in $\rkl$ because $(1^i)$ has $i>k$ rows, so
$(1^i)\not\subseteq k\times\ell$. Hence for any $m$ with $k\le m\le\ell$ we have
$\Q\langle e_1,\dots,e_m\rangle=\Q\langle e_1,\dots,e_k\rangle=\rkl$, i.e.\
$\Rklm=R^{k,\ell,k}=\rkl$ (this also reproves Lemma~\ref{lem:trivial} for the
range $m\ge k$). In particular, for $m>k$ both $\Rklm=\rkl$ and
$R^{k,\ell,m-1}=\rkl$ (as $m-1\ge k$ too), so
$Q_{k,\ell,m}=\Hilb(\rkl/\rkl)=0=F_{k,\ell,m}$. The formula \eqref{eq:main} thus
reproduces the degenerate behaviour.
\end{proof}

\begin{lemma}[The case $m=1$]\label{lem:m1}
For all $k\ge1$ and $\ell\ge1$,
\[
F_{k,\ell,1}(q)=q\,[\ell]_q\sum_{j=0}^{k-1}q^{\,j\ell}
=q\,[\ell]_q\,[k]_{q^{\ell}}
=\frac{q(1-q^{\ell})(1-q^{k\ell})}{(1-q)(1-q^{\ell})}
=\frac{q\,(1-q^{k\ell})}{1-q}.
\]
Equivalently $Q_{k,\ell,1}=\Hilb\big(\Q[e_1]/\Q;q\big)$ is the Hilbert series of the
quotient of the polynomial-image subalgebra $R^{k,\ell,1}=\Q\langle e_1\rangle$ by
its degree-$0$ part.
\end{lemma}

\begin{proof}
Set $m=1$ in $F_{k,\ell,m}$. Then $\qbin{\ell}{1}=[\ell]_q$, the inner exponent is
$j(\ell-1+1)=j\ell$, and $\qbin{1+j-1}{j}=\qbin{j}{j}=1$ for every $j$. Hence
\[
F_{k,\ell,1}=q\,[\ell]_q\sum_{j=0}^{k-1}q^{\,j\ell}
=q\,[\ell]_q\cdot\frac{1-q^{k\ell}}{1-q^{\ell}}
=\frac{q\,(1-q^{\ell})}{1-q}\cdot\frac{1-q^{k\ell}}{1-q^{\ell}}
=\frac{q(1-q^{k\ell})}{1-q},
\]
which is the asserted closed form, with $[k]_{q^{\ell}}=\sum_{j=0}^{k-1}q^{j\ell}$.
Finally $R^{k,\ell,1}=\Q\langle e_1\rangle$ is generated by the single element
$\overline{e_1}$ of degree $1$, and $\frac{q(1-q^{k\ell})}{1-q}=q+q^2+\dots+q^{k\ell}$
records that $\overline{e_1^{\,d}}\ne0$ for $1\le d\le k\ell=\dim_{\C}\Gr(k,\C^{k+\ell})$
and $\overline{e_1^{\,d}}=0$ beyond the top degree; this is consistent with $e_1$
being the hyperplane (Schubert) class, whose top nonzero power is the fundamental
class in degree $k\ell$. (We use this only as a check; the main theorem already
covers $m=1$.)
\end{proof}

We now turn to the telescope.

\begin{theorem}[Global telescope]\label{thm:telescope}
For all $k\ge1$ and $\ell\ge1$,
\begin{equation}\label{eq:telescope}
\sum_{m=1}^{\ell}F_{k,\ell,m}(q)=\qbin{k+\ell}{k}-1 .
\end{equation}
Equivalently, granting the main theorem \eqref{eq:main},
$\sum_{m=1}^{\ell}Q_{k,\ell,m}(q)=\Hilb(\rkl;q)-1$.
\end{theorem}

\begin{proof}[Proof as a corollary of \eqref{eq:main}]
The subalgebras form an increasing chain
$\Q=R^{k,\ell,0}\subseteq R^{k,\ell,1}\subseteq\dots\subseteq R^{k,\ell,\ell}=\rkl$,
the last equality by Lemma~\ref{lem:trivial} (the case $m=\ell$, valid for all
$k\ge1$). Since each inclusion is of graded subspaces,
$Q_{k,\ell,m}=\Hilb(\Rklm;q)-\Hilb(R^{k,\ell,m-1};q)$, and the sum telescopes:
\[
\sum_{m=1}^{\ell}Q_{k,\ell,m}
=\Hilb(R^{k,\ell,\ell};q)-\Hilb(R^{k,\ell,0};q)
=\Hilb(\rkl;q)-1=\qbin{k+\ell}{k}-1,
\]
using \eqref{eq:HilbR}. By the main theorem $Q_{k,\ell,m}=F_{k,\ell,m}$, which gives
\eqref{eq:telescope}.
\end{proof}

The above derivation is logically dependent on \eqref{eq:main}. We now give a
self-contained proof of the pure $q$-identity \eqref{eq:telescope}, independent of
the cohomological content; this both serves as a strong check on $F_{k,\ell,m}$ and
shows \eqref{eq:telescope} is a genuine corollary of the closed form alone.

\begin{lemma}[Single-step identity]\label{lem:step}
Write $F_{k,\ell,m}=0$ for $m>k$ \textup{(}empty inner sum,
Lemma~\ref{lem:mbigk}\textup{)} and $F_{0,\ell,m}:=0$. For every $k\ge1$ and
$\ell\ge1$,
\begin{equation}\label{eq:stepid}
\sum_{m=1}^{\ell}\big(F_{k,\ell,m}-F_{k-1,\ell,m}\big)
=\qbin{k+\ell}{k}-\qbin{k+\ell-1}{k-1}.
\end{equation}
\end{lemma}

\begin{proof}
Fix $\ell$ and $k\ge1$. By Lemma~\ref{lem:RHSrec}, for $1\le m<k$ (necessarily
$m\le\ell$ in the sum) we have
\[
F_{k,\ell,m}-F_{k-1,\ell,m}=q^{\,m+(k-m)(\ell-m+1)}\qbin{\ell}{m}\qbin{k-1}{m-1}.
\]
For $m=k$ (which contributes only when $k\le\ell$), $F_{k-1,\ell,k}=0$ since $k>k-1$,
and by \eqref{eq:RHSbase} $F_{k,\ell,k}=q^{k}\qbin{\ell}{k}$; this equals the same
expression evaluated at $m=k$, because then $(k-m)(\ell-m+1)=0$ and
$\qbin{k-1}{m-1}=\qbin{k-1}{k-1}=1$. For $m>k$ both terms vanish, and also the
expression $q^{\,m+(k-m)(\ell-m+1)}\qbin{\ell}{m}\qbin{k-1}{m-1}$ vanishes there
because $\qbin{k-1}{m-1}=0$ for $m-1>k-1$. Likewise for $k\le m\le\ell$ with
$m>k$ excluded the bookkeeping is consistent. Hence, extending the sum harmlessly
to all $m\ge1$ (every term with $m>\min(k,\ell)$ is $0$, as $\qbin{\ell}{m}=0$ for
$m>\ell$ and $\qbin{k-1}{m-1}=0$ for $m>k$),
\begin{equation}\label{eq:Dunified}
\sum_{m=1}^{\ell}\big(F_{k,\ell,m}-F_{k-1,\ell,m}\big)
=\sum_{m\ge1}q^{\,m+(k-m)(\ell-m+1)}\qbin{\ell}{m}\qbin{k-1}{m-1}=:D_k.
\end{equation}
Substitute $i=m-1\ge0$. Using
$m+(k-m)(\ell-m+1)=k+(i-k+1)(i-\ell+1)=k+(k-1-i)(\ell-1-i)$ (an elementary
expansion: both sides equal $k\ell-k m-\ell m+m^2+k$ after $m=i+1$), we get
\[
D_k=q^{k}\sum_{i\ge0}q^{(k-1-i)(\ell-1-i)}\qbin{\ell}{i+1}\qbin{k-1}{i}.
\]
Now apply the $q$-Vandermonde (Gaussian convolution) identity in the form
\begin{equation}\label{eq:qvander}
\qbin{M+N}{r}=\sum_{j\ge0}q^{(r-j)(M-j)}\qbin{M}{j}\qbin{N}{r-j},
\end{equation}
valid for all nonnegative integers $M,N,r$ (a standard $q$-binomial theorem
identity; see e.g.\ Andrews, \emph{The Theory of Partitions}, Ch.~3). Take
$M=\ell$, $N=k-1$, $r=k$. Then the right-hand side of \eqref{eq:qvander} is
\[
\sum_{j\ge0}q^{(k-j)(\ell-j)}\qbin{\ell}{j}\qbin{k-1}{k-j}
=\sum_{j\ge1}q^{(k-j)(\ell-j)}\qbin{\ell}{j}\qbin{k-1}{j-1},
\]
where we used $\qbin{k-1}{k-j}=\qbin{k-1}{(k-1)-(k-j)}=\qbin{k-1}{j-1}$ and dropped
the $j=0$ term ($\qbin{k-1}{k}=0$). Reindexing $i=j-1$ gives exactly
$q^{-k}D_k$, since $(k-j)(\ell-j)=(k-1-i)(\ell-1-i)$. Hence
$D_k=q^{k}\qbin{k+\ell-1}{k}$. Finally, the $q$-Pascal rule
$\qbin{k+\ell}{k}=\qbin{k+\ell-1}{k-1}+q^{k}\qbin{k+\ell-1}{k}$ yields
$q^{k}\qbin{k+\ell-1}{k}=\qbin{k+\ell}{k}-\qbin{k+\ell-1}{k-1}$, which is the
right-hand side of \eqref{eq:stepid}. This proves \eqref{eq:stepid}.
\end{proof}

\begin{proof}[Self-contained proof of Theorem~\ref{thm:telescope}]
Sum \eqref{eq:stepid} over $k=1,2,\dots,K$. The left-hand side telescopes in $k$:
\[
\sum_{k=1}^{K}\sum_{m=1}^{\ell}\big(F_{k,\ell,m}-F_{k-1,\ell,m}\big)
=\sum_{m=1}^{\ell}\big(F_{K,\ell,m}-F_{0,\ell,m}\big)
=\sum_{m=1}^{\ell}F_{K,\ell,m},
\]
using $F_{0,\ell,m}=0$. The right-hand side telescopes to
$\qbin{K+\ell}{K}-\qbin{\ell}{0}=\qbin{K+\ell}{K}-1$. Renaming $K$ as $k$ gives
\eqref{eq:telescope}.
\end{proof}

\begin{remark}[Numerical confirmation]\label{rem:telnum}
Lemmas~\ref{lem:mbigk},~\ref{lem:m1},~\ref{lem:step} and
Theorem~\ref{thm:telescope} were verified symbolically (exact rational functions in
$q$) for all $1\le k,\ell\le5$ and $1\le m\le\ell$: $F_{k,\ell,m}=0$ for $m>k$, the
$m=1$ closed form $q(1-q^{k\ell})/(1-q)$, the single-step identity \eqref{eq:stepid}
$=q^k\qbin{k+\ell-1}{k}$, and the telescope $\sum_m F_{k,\ell,m}=\qbin{k+\ell}{k}-1$.
All agreed identically.
\end{remark}

\section*{Remaining gap}

The reduction above is rigorous and complete: \eqref{eq:main} is equivalent to the
pair \eqref{eq:LHSrec}--\eqref{eq:LHSbase}; the surjection $\pi$ and its kernel
$e_k\rkl$ are established (Lemma~\ref{lem:pi}); $\pi$ respects the filtration
(Corollary~\ref{cor:pifiltr}); the right-hand side recursion is proved
(Lemma~\ref{lem:RHSrec}); and the cases $m\ge k$, $m=\ell$ are handled
(Lemma~\ref{lem:trivial}). The boundary and global-consistency requirements are
fully discharged in Section~\ref{sec:boundary}: the degenerate range $m>k$ (where
both sides vanish, Lemma~\ref{lem:mbigk}), the case $m=1$
(Lemma~\ref{lem:m1}), and the global telescope
$\sum_{m=1}^{\ell}F_{k,\ell,m}=\qbin{k+\ell}{k}-1$
(Theorem~\ref{thm:telescope}), the last derived both as a corollary of
\eqref{eq:main} and, independently, from the closed form via $q$-Vandermonde
(Lemma~\ref{lem:step}). We have additionally shown
(Section~\ref{sec:Fm}) that the natural column-height span $F_m$ is \emph{not} a
viable route: $\Rklm\not\subseteq F_m$ in general (Lemma~\ref{lem:RnotinF}), and $F_m\not\subseteq\Rklm$ (Lemma~\ref{lem:FnotinR}), so the
subalgebra has \emph{no} Schur-class basis cut out by a column-height bound, and the
inductive route through $\pi$ is essential.

What is not yet derived from first principles is the
Hilbert series \eqref{eq:kernelHilb} of $\ker\overline\pi$ (the inductive step) and
the complementary count \eqref{eq:basecum} (the base case $k=m$). Both have been
verified symbolically for all $1\le k,\ell\le5$, $1\le m\le\ell$
(Remark~\ref{rem:verif}); supplying these two computations from the algebra of
$\rkl$ completes the proof of \eqref{eq:main}.
