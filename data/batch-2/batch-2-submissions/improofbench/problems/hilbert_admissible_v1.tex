\textbf{Problem.} Fix positive integers $n$ and $d$ and consider the
Hilbert scheme of $n$ points on affine $d$-space,
\[
  X = \mathrm{Hilb}_n(\mathbb{A}^d_{\mathbb{C}}),
\]
parametrizing zero-dimensional closed subschemes
$Z \subset \mathbb{A}^d_{\mathbb{C}}$ of length $n$.

\medskip

\noindent\textbf{Definition.} A point $[Z] \in X$ is called
\emph{admissible} if both of the following hold:
\begin{enumerate}
  \item $Z$ is set-theoretically supported on a single point of
        $\mathbb{A}^d_{\mathbb{C}}$;
  \item there exists an analytic open neighborhood $U \subseteq X$ of
        $[Z]$ such that for every $[Z'] \in U$ the structure sheaf
        $\mathcal{O}_{Z'}$ (equivalently, the global section algebra
        $H^0(Z', \mathcal{O}_{Z'})$) is isomorphic to $\mathcal{O}_Z$
        as a $\mathbb{C}$-algebra.
\end{enumerate}

\medskip

\noindent\textbf{Question.} Does there exist a pair $(n, d)$ with
$n > 1$ such that the Hilbert scheme
$X = \mathrm{Hilb}_n(\mathbb{A}^d_{\mathbb{C}})$ contains an admissible
point?

\smallskip

Either:
\begin{itemize}
  \item exhibit explicit values of $n$ and $d$ together with an
        explicit admissible point $[Z] \in X$, and prove that it is
        admissible; or
  \item prove that no admissible point exists for any $n > 1$ and any
        $d \ge 1$.
\end{itemize}

\bigskip

\noindent\textbf{Constraints.}
\begin{itemize}
  \item Final answer must be a self-contained mathematical document.
  \item Maximum length: 35 pages of standard \texttt{article} class
        output.
\end{itemize}
