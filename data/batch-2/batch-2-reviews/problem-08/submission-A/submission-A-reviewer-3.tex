\subsection*{0. Conventions and two local facts}

Let $E$ be the ground set, $r=\rk M$, and $\mu_M$ the trivial rank $r$
valuated matroid supported on the bases of $M$.  I use the standard
model of the relative Dressian: a rank $k$ point is a projective
valuated matroid $\nu$ with $\nu\preceq\mu_M$, where $\preceq$ denotes
valuated quotient.  Brandt--Eur--Zhang \cite{BEZ} identify this order with
inclusion of projective tropical linear spaces,
\[
        \Trop(\theta)\subseteq\Trop(\nu)
        \quad\Longleftrightarrow\quad \theta\preceq\nu .
\]
For valuated matroids $p,q$ of ranks $a\le b$, $p\preceq q$ is
equivalent to the incidence-Plucker relations
\[
 \min_{i\in J\setminus I}\{p(I\cup i)+q(J\setminus i)\}
 \quad\hbox{attained at least twice}                         \tag{IP}
\]
for all $I\in\binom{E}{a-1}$ and $J\in\binom{E}{b+1}$.  All-$\infty$
minima are regarded as attained at every displayed index.

\begin{criterion}[nested incidence suffices]\label{crit:local}
Let $p,q$ be valuated matroids of ranks $a<b$ whose ordinary supports
are in quotient order.  If {\rm (IP)} holds whenever $I\subseteq J$,
then {\rm (IP)} holds for all $I,J$.
\end{criterion}

\begin{proof}
For a rank $a$ valuated matroid $p$ and an $(a-1)$-set $I$, write
$C^*_p(I)_e=p(I\cup e)$, with value $\infty$ if $e\in I$.  We use the
standard cocircuit-generation theorem of Dress--Wenzel/Speyer \cite{DW,Speyer}: the
tropical linear space of a valuated matroid is the closed tropical convex
hull of its valuated cocircuits.

Fix arbitrary $I\in\binom{E}{a-1}$ and $J\in\binom{E}{b+1}$, and put
$c_j=q(J\setminus j)$ for $j\in J$.  If all $c_j$ are $\infty$ there is
nothing to prove.  Otherwise $J$ contains a $q$-basis.  Since the
ordinary support of $p$ is a quotient of the ordinary support of $q$,
every $q$-basis spans the support matroid of $p$ (indeed,
$a-\rk_p(Q)\le b-\rk_q(Q)=0$ for a $q$-basis $Q$); hence the
restriction $p|J$ has rank $a$.

For every $S\subseteq J$ with $|S|=a-1$, the nested hypothesis with
upper set $J$ says precisely that the cocircuit $C^*_{p|J}(S)$ is
tropically orthogonal to $c=(c_j)_{j\in J}$.  The set of vectors
orthogonal to a fixed $c$ is a tropical hyperplane, hence is tropically
convex; therefore it contains the whole tropical linear space of $p|J$.
It remains only to note that $C^*_p(I)|_J$ is a vector of $p|J$: for
each valuated circuit of $p|J$, indexed by a set
$D\in\binom{J}{a+1}$, the orthogonality relation
\[
 \min_{e\in D}\{p(I\cup e)+p(D\setminus e)\}
\]
is exactly a Plucker relation for $p$.  Hence $C^*_p(I)|_J$ is
orthogonal to $c$, which is the desired arbitrary incidence relation.
\end{proof}

\begin{proposition}[elementary quotient test]\label{prop:elementarytest}

$\theta:\binom{E}{d-1}\to{\mathbb R}\cup\{\infty\}$ be nonzero.  Suppose
that
\[

        \hbox{ is dependent in the support matroid of }\eta              \tag{D}
\]
and, when $d\ge2$, for every $S\in\binom{E}{d-2}$ and distinct $x,y,z\notin S$,
\[
\min\{\theta(Sx)+\eta(Syz),\,
      \theta(Sy)+\eta(Sxz),\,
      \theta(Sz)+\eta(Sxy)\}                                      \tag{C}
\]
is attained at least twice.  Then $\theta$ is a valuated matroid and
$\theta\preceq\eta$.
\end{proposition}

\begin{proof}
For $d=1$ this is the unique rank-zero quotient.  Let $d\ge2$.  By
Wang's Theorem 4.3 \cite{Wang}, $\Dr^1(\eta)$ is cut out by the
degenerate, parallel, and concurrent three-term incidence relations.
Condition (D) is exactly the degenerate part, and (C) contains the
concurrent part.  It remains only to see that the parallel part follows
from (C).  Write $A=Di$ and $B=Dj$ for independent $(d-1)$-sets in the
support matroid of $\eta$ with the same closure, and choose $k$ outside
that closure, so $Dik$ and $Djk$ are bases.  Applying (C) to
$D,i,j,k$, the third upper coordinate $\eta(Dij)$ is $\infty$, while
$\eta(Dik)$ and $\eta(Djk)$ are finite.  Hence
\[
        \theta(Di)+\eta(Djk)=\theta(Dj)+\eta(Dik),
\]
which is Wang's parallel hyperplane relation.  Thus all of Wang's
defining relations hold, so $\theta$ represents a codimension-one
tropical linear subspace of $\Trop(\eta)$.
\end{proof}

\subsection*{1. Flat-lattice duality and flat constancy}

\begin{lemma}\label{lem:modular}
If $\cL(M)$ admits an order-reversing involution $X\mapsto X^\perp$,
then $\cL(M)$ is modular and $\rk(X^\perp)=r-\rk(X)$.
\end{lemma}

\begin{proof}
The anti-automorphism reverses maximal chains, so it sends rank $i$ to
rank $r-i$.  Applying it to the semimodular inequality for the geometric
lattice $\cL(M)$ gives the reverse inequality, hence equality.
\end{proof}

\begin{lemma}[flat constancy]\label{lem:flatconst}
Let $\nu\preceq\mu_M$ have rank $k$.  If $B,B'$ are independent
$k$-subsets of $M$ with $\cl(B)=\cl(B')$, then $\nu(B)=\nu(B')$; and
$\nu(A)=\infty$ whenever $A$ is dependent in $M$.
\end{lemma}

\begin{proof}
The support matroid of $\nu$ is an ordinary quotient of $M$, so its
bases are independent in $M$.  For $k>0$ it suffices to compare
$B=A\cup b$ and $B'=A\cup b'$ spanning the same rank $k$ flat.  Choose
one complement $C$ so that both $AbC$ and $Ab'C$ are $M$-bases.  In the
incidence relation for $\nu\preceq\mu_M$ with lower set $A$ and upper
set $A\cup\{b,b'\}\cup C$, the only finite $\mu_M$-terms are the
$b$- and $b'$-terms.  Thus $\nu(Ab)=\nu(Ab')$.
\end{proof}
\review{Lemmas 1 and 2 have the correct degree of concision. Still nicely explained.}


Write $\bar\nu(X)$ for the common value on independent $k$-sets spanning
the rank $k$ flat $X$, and put $\bar\nu(X)=\infty$ if no basis of $\nu$
spans $X$.

\subsection*{2. The polar transform}

For a rank $k$ quotient $\nu\preceq\mu_M$ define
\[
 \nu^{\perp_M}(A)=
 \begin{cases}
   \bar\nu\bigl((\cl A)^\perp\bigr),
      & A\text{ independent in }M,\ |A|=r-k,\\
   \infty,&\text{otherwise.}
 \end{cases}                                                 \tag{2}
\]
This is projectively well-defined, and applying it twice recovers the
original flat coordinates.

\review{This is the correct definition. It should be made clearer that $\cl$ means the closure in $M$.}

\begin{lemma}[ordinary polar rank]\label{lem:ordinary}
Let $N$ be an ordinary quotient of $M$ of rank $k$, with rank function
$\rho_N$.  Then
\[
        \rho^\perp(A)=\rk_M(A)-k+\rho_N((\cl A)^\perp)          \tag{3}
\]
is a matroid rank function.  Its bases are precisely the finite support
of (2).  If $N_p$ is a quotient of $N_q$, then
$N_q^{\perp_M}$ is a quotient of $N_p^{\perp_M}$.
\end{lemma}

\begin{proof}
Since $N$ is a quotient of $M$, $\rho_N$ is constant on $M$-closures:
if $e\in\cl_M(A)$, then the $N$-rank increment from $A$ to $Ae$ is at
most the corresponding $M$-increment, namely $0$.  First restrict (3) to
flats.  For flats $X\le Y$,
\[
\rho^\perp(Y)-\rho^\perp(X)
 =\rk(Y)-\rk(X)-\bigl(\rho_N(X^\perp)-\rho_N(Y^\perp)\bigr)
\]
is between $0$ and $\rk(Y)-\rk(X)$ by the strong-map increment
inequality, so $\rho^\perp$ is monotone with unit increments on flats.
For flats $X,Y$, modularity of $\rk_M$ and the identities
$(X\wedge Y)^\perp=X^\perp\vee Y^\perp$ and
$(X\vee Y)^\perp=X^\perp\wedge Y^\perp$ reduce submodularity of
$\rho^\perp$ to submodularity of $\rho_N$.

For arbitrary subsets, $\rho^\perp(A)=\rho^\perp(\cl_M A)$.  If
$X=\cl A$ and $Y=\cl B$, then
$\cl(A\cap B)\le X\wedge Y$ and $\cl(A\cup B)=X\vee Y$; by monotonicity
on flats, submodularity for $X,Y$ implies submodularity for $A,B$.
The unit-increment property also passes from flats to subsets.  Thus
(3) is a matroid rank function.  An $M$-independent set $A$ of size
$r-k$ has full $\rho^\perp$-rank exactly when
$\rho_N((\cl A)^\perp)=k$, and $M$-dependent sets cannot be bases; this
is precisely the finite support of (2).

If $N_p$ is a quotient of $N_q$, then on every interval the rank
increment of $N_p$ is at most that of $N_q$.  The displayed formula for
increments therefore reverses the inequality for the polar ranks, which
is exactly the strong-map criterion saying that $N_q^{\perp_M}$ is a
quotient of $N_p^{\perp_M}$.
\end{proof}

\begin{lemma}[elementary opposition]\label{lem:elementary}
Let $p\preceq q\preceq\mu_M$, with $\rk p=d$ and $\rk q=d+1$.  Then
$q^{\perp_M}$ and $p^{\perp_M}$ satisfy all adjacent three-term
relations of ranks $r-d-1$ and $r-d$.
\end{lemma}

\begin{proof}
Fix $A$ of size $r-d-2$ and distinct $x,y,z\notin A$.  Put
$U=\cl(A)$, $W=\cl(Axyz)$, and $\delta=\rk W-\rk U$.  If $A$ is
dependent, or if $\delta\le1$, each term contains a dependent set of the
required size and hence is $\infty$.

If $\delta=2$, let $G=W^\perp$ and $H=U^\perp$, of ranks $d$ and
$d+2$.  Whenever the term indexed by $x$ is finite, $Ayz$ spans $W$,
so its $p$-factor is the common value $\bar p(G)$; its $q$-factor is
$\bar q((\cl(Ax))^\perp)$, and similarly for $y,z$.  In the rank-two
interval $[U,W]$, a loop over $U$ makes all terms involving it
infinite, and two parallel elements give equal corresponding
$q$-coordinates.  Thus only the case of three non-parallel points over
$U$ needs proof.  Choose a basis $S$ of $G$, representatives
$a_x,a_y,a_z$ for the three dual rank-one flats over $G$ in $[G,H]$,
and a complement $C$ of $H$ to a basis of $M$.  In the incidence
relation for $q\preceq\mu_M$ with lower set $S$ and upper set
$S\cup\{a_x,a_y,a_z\}\cup C$, the only finite $\mu_M$-terms are those
obtained by deleting one of $a_x,a_y,a_z$.  Hence the three displayed
$q$-coordinates have a non-unique minimum; adding the common
$\bar p(G)$ proves the transformed relation.

If $\delta=3$, then $x,y,z$ form a Boolean frame over $U$.  Let
$G=W^\perp$ and
\[
 P_x=(\cl(Ayz))^\perp,\quad P_y=(\cl(Axz))^\perp,\quad
 P_z=(\cl(Axy))^\perp .
\]
These are the three atoms over $G$ in the dual frame, and
$(\cl(Ax))^\perp=P_y\vee P_z$, with the analogous identities for
$y,z$.  Choose a basis $S$ of $G$ and representatives
$a_e\in P_e\setminus G$.  The adjacent incidence relation for
$p\preceq q$ applied to the lower set $S$ and to
$S\cup\{a_x,a_y,a_z\}$ is exactly the transformed three-term relation,
using flat constancy to identify the displayed coordinates.
\end{proof}
\review{In the above lemma, once gets the sense that there is a picture being drawn that is not shown to the reader. Either a physical depiction and/or writing down the actual equations would make the argument more palatable. This proof is correct but the reasoning is obscured by the failure to highlight the facts that make the argument work.}

\begin{proposition}\label{prop:set}
The formula $\nu\mapsto\nu^{\perp_M}$ defines an involution of the set
underlying $\Dr(M)$.
\end{proposition}

\begin{proof}
Insert $\nu$ in the complete flag (1).  We prove inductively that
\[
        h_r^{\perp_M}\preceq h_{r-1}^{\perp_M}\preceq\cdots
        \preceq h_0^{\perp_M}=\mu_M .
\]
After harmless scalar normalization, the right endpoint
$h_0^{\perp_M}=\mu_M$ is clear from (2).  If $r\le1$, there is
nothing more to check beyond the automatic rank-zero quotient.  Now assume $r\ge2$.
Suppose $0\le j\le r-2$ and $h_j^{\perp_M}$ has
already been proved valuated.  Since $h_j\preceq h_{j+1}$, Lemma
\ref{lem:ordinary} says that the ordinary support of
$h_{j+1}^{\perp_M}$ is an elementary quotient of the ordinary support of
$h_j^{\perp_M}$; in particular it satisfies the degenerate support
condition (D) relative to $h_j^{\perp_M}$.  Lemma
\ref{lem:elementary} gives the displayed concurrent relations (C) for
the pair $h_{j+1}^{\perp_M},h_j^{\perp_M}$.  Proposition
\ref{prop:elementarytest} then implies
$h_{j+1}^{\perp_M}\preceq h_j^{\perp_M}$.  This proves the induction
down to $h_{r-1}^{\perp_M}$; the last rank-zero quotient
$h_r^{\perp_M}\preceq h_{r-1}^{\perp_M}$ is automatic.  Hence $\nu^{\perp_M}$, one layer of the flag, lies in
$\Dr(M)$.  Involutivity follows from (2) and $(X^\perp)^\perp=X$.
\end{proof}

\subsection*{3. Order reversal}

\begin{lemma}[circuit minima]\label{lem:circuit}

$(k-1)$-set of $M$, and let $C$ be a circuit of $M/A$ with $|C|\ge2$.

\end{lemma}

Let $F=\cl(A\cup C)$.  For every $i\in C$, $A\cup(C-i)$ is independent

the incidence relation for $w\preceq\mu_M$ with lower set $A$ and upper
set $A\cup C\cup D$, the finite $\mu_M$-terms are exactly those indexed
by $i\in C$.
\end{proof}
\review{The above lemma is completely trivial and standard. It should be stated, not proven.}

\begin{theorem}\label{thm:order}
If $p\preceq q\preceq\mu_M$, then
\[
        q^{\perp_M}\preceq p^{\perp_M}.
\]
\end{theorem}

\begin{proof}
Let $\rk p=a$ and $\rk q=b$.  If $a=b$, the strong-map inequalities applied to
$\emptyset\subseteq A$ and to $A\subseteq E$ force the two ordinary
support rank functions to be equal.  Then $p$ and $q$ are projectively
equal: the equal-rank incidence relation on adjacent bases shows
$p(B)-q(B)$ is constant on the common basis graph.  If $a=0$, then $p^{\perp_M}=\mu_M$, so the
assertion is $q^{\perp_M}\preceq\mu_M$; if $b=r$, it is the vacuous
rank-zero lower quotient.  Hence assume $0<a<b<r$.

By Proposition \ref{prop:set} the polars are valuated matroids, and by
Lemma \ref{lem:ordinary} their ordinary supports are in quotient order.
By Criterion \ref{crit:local} it remains to prove nested incidence.
Fix $A\subseteq T$ with
$|A|=r-b-1$, $|T|=r-a+1$, set $B=T\setminus A$, and
$m=|B|=b-a+2$.  If $A$ is dependent all terms are $\infty$.  Otherwise
put $U=\cl(A)$, $W=\cl(T)$, and $\delta=\rk W-\rk U$.  A finite term
forces $T-i$ to be independent, hence $\delta\ge m-1$; since
$\delta\le m$, only two cases remain.

If $\delta=m$, then $T$ is independent over $U$.  Let $G=W^\perp$ and
$H=U^\perp$, of ranks $a-1$ and $b+1$.  The interval generated by $B$
is Boolean; after applying $^\perp$, the flats
$X_i=(\cl(T-i))^\perp$ are the atoms over $G$, and
$Y_i=(\cl(Ai))^\perp$ is the join of all $X_j$ with $j\ne i$.  Choose a
basis $S$ of $G$ and representatives $x_i\in X_i\setminus G$.  The
nested incidence relation for $p\preceq q$ with lower set $S$ and upper
set $S\cup\{x_i:i\in B\}$ is exactly the desired transformed relation.

If $\delta=m-1$, then $B$ has nullity one in $M/A$, hence a unique

and spans $W$, so the $p^{\perp_M}(T-i)$ factor is the common value

\end{proof}

\subsection*{4. Agreement with flats}

Let $F$ be a flat of rank $f$, and let $q_F$ be the trivial rank
$r-f$ valuated matroid of $M/F\oplus U_{0,F}$.  For an independent
$(r-f)$-set $B$ with $Z=\cl(B)$,
\[
        q_F(B)<\infty\quad\Longleftrightarrow\quad Z\vee F=\hat1.
\]
For an independent $f$-set $A$ with $Y=\cl(A)$, formula (2) gives
\[
 q_F^{\perp_M}(A)<\infty
 \quad\Longleftrightarrow\quad Y^\perp\vee F=\hat1 .
\]
By modularity, since $\rk(Y^\perp)=r-f$ and $\rk F=f$, this is
equivalent to $Y^\perp\wedge F=\hat0$.  Applying the anti-automorphism
gives $Y\vee F^\perp=\hat1$, the basis condition for
$M/F^\perp\oplus U_{0,F^\perp}$.  The finite values are zero on both
sides.  Thus the polar transform sends the flat $F$ to $F^\perp$.

Combining Proposition \ref{prop:set}, Theorem \ref{thm:order}, and this
flat computation, the map
\[
        \Phi([\nu])=[\nu^{\perp_M}]
\]
is an order-reversing involution of $\Dr(M)$ extending the given
order-reversing involution of $\cL(M)$.
