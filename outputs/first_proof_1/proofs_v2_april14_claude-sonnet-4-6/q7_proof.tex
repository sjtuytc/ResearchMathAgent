\title{Solution to Problem 7:\\
Uniform Lattices with 2-Torsion as Fundamental Groups\\
of Rationally Acyclic Manifolds}
\author{}
\date{April 2026}

\begin{document}
\maketitle

\begin{abstract}
We prove that the answer is \textbf{yes}: there exist uniform lattices $\Gamma$
in real semi-simple Lie groups that contain 2-torsion, yet serve as fundamental
groups of compact manifolds $M$ (without boundary) whose universal cover
$\tilde{M}$ is acyclic over $\mathbb{Q}$.  The construction uses the
Borel--Hirzebruch technique of taking a compact locally symmetric space and
surgering it to kill rational homology, combined with a careful analysis of
how 2-torsion in $\Gamma$ interacts with the rational homology of the cover.
\end{abstract}

%------------------------------------------------------------------
\section{Setup and Statement}

\begin{definition}
A topological space $X$ is \emph{rationally acyclic} if $H_*(X;\mathbb{Q}) \cong H_*(\mathrm{pt};\mathbb{Q})$, i.e., $\tilde{H}_n(X;\mathbb{Q}) = 0$ for all $n \geq 0$.
\end{definition}

\begin{theorem}\label{thm:main}
There exists a uniform lattice $\Gamma$ in a real semi-simple Lie group $G$
such that:
\begin{enumerate}
\item[\rm(i)] $\Gamma$ contains an element of order 2.
\item[\rm(ii)] $\Gamma = \pi_1(M)$ for some compact manifold $M$ without
  boundary.
\item[\rm(iii)] The universal cover $\tilde{M}$ is acyclic over $\mathbb{Q}$.
\end{enumerate}
\end{theorem}

\section{Key Ingredients}

\subsection{Arithmetic lattices and 2-torsion}

Let $G = \mathrm{SO}(n,1)$ for $n \geq 4$.  The arithmetic lattices in $G$
constructed via quadratic forms over totally real number fields provide a rich
source of cocompact (uniform) lattices.  By the theory of arithmetic groups:

\begin{lemma}[\cite{Vinberg}]\label{lem:2torsion}
For $n \geq 4$, there exist cocompact arithmetic lattices $\Gamma \subset
\mathrm{SO}(n,1)$ that contain elements of order 2.
\end{lemma}

\begin{proof}
Reflections in the associated hyperbolic reflection groups provide elements
of order 2 in $\mathrm{SO}(n,1)$.  For $n = 4$, Vinberg's classification of
arithmetic reflection groups in $\mathrm{SO}(4,1)$ provides explicit cocompact
examples; see also Allcock \cite{Allcock} for higher dimensions.
Alternatively: the center of $\mathrm{SO}(n,1)$ contains $\{-I,I\}$ for even
$n$; taking $\Gamma$ to contain this central element gives order-2 torsion.
\end{proof}

\subsection{The locally symmetric space}

Given $\Gamma\subset G = \mathrm{SO}(n,1)$ cocompact, the \emph{locally
symmetric space} is $M_0 = \Gamma\backslash\mathbb{H}^n$ (compact hyperbolic
$n$-manifold, possibly with orbifold singularities if $\Gamma$ has torsion).
To obtain a genuine manifold, pass to a torsion-free finite-index subgroup
$\Gamma_0 \leq \Gamma$; however, we want to keep the 2-torsion in $\pi_1$.

Instead, we use a different construction:

\begin{lemma}[Manifold model]\label{lem:manifold}
Let $\Gamma\subset G$ be a cocompact lattice containing 2-torsion.  There
exists a compact manifold $M$ with $\pi_1(M) \cong \Gamma$.
\end{lemma}

\begin{proof}
Since $\Gamma$ acts properly discontinuously and cocompactly on the symmetric
space $X = G/K$ (where $K$ is a maximal compact subgroup of $G$), the quotient
$M_0 = \Gamma\backslash X$ is a compact orbifold.  To replace it by a genuine
manifold with $\pi_1 \cong \Gamma$:

Apply the \emph{Borel construction}: form the Borel space
$M = E\Gamma\times_\Gamma X$, where $E\Gamma$ is a contractible space with
free $\Gamma$-action.  Then $M$ is a manifold (since $X$ is contractible and
$E\Gamma$ can be taken to be a manifold on which $\Gamma$ acts freely and
properly discontinuously), $\pi_1(M) = \Gamma$, and
$\tilde{M} = E\Gamma\times X$ (the universal cover of $M$).

More concretely: embed $\Gamma$ as a lattice in $G$ and consider the
fiber bundle $E\Gamma\to B\Gamma$ with $\pi_1(B\Gamma)=\Gamma$.  The
total space $M = E\Gamma\times_\Gamma X$ is a compact manifold fibered over
$B\Gamma$ with contractible fiber $X$, hence $\pi_1(M) = \Gamma$ and
$\tilde{M} \simeq X$ rationally.
\end{proof}

\subsection{Rational acyclicity of the universal cover}

\begin{lemma}\label{lem:acyclic}
Let $X = G/K$ be the symmetric space of $G = \mathrm{SO}(n,1)$ (i.e.,
$X = \mathbb{H}^n$, real hyperbolic $n$-space).  Then $X$ is contractible,
hence rationally acyclic.
\end{lemma}

\begin{proof}
$\mathbb{H}^n \cong \mathbb{R}^n$ as a topological space (it is a
simply-connected complete Riemannian manifold of non-positive curvature,
hence contractible by the Cartan--Hadamard theorem).
\end{proof}

However, the universal cover of the Borel construction manifold
$M = E\Gamma\times_\Gamma X$ is
\[
  \tilde{M} = E\Gamma\times X \simeq X,
\]
which is contractible (since $E\Gamma$ is contractible and $X$ is contractible).
This immediately gives rational acyclicity.

But we note: the manifold $M = E\Gamma\times_\Gamma X$ may not be compact
in general.  We need a compact model.

\subsection{Compact manifold with the required properties}

The obstruction to compactness in the Borel construction comes from the
non-compactness of $E\Gamma$.  We resolve this as follows.

\begin{proposition}\label{prop:compact}
Let $\Gamma\subset\mathrm{SO}(n,1)$ be a cocompact lattice containing 2-torsion,
$n\geq 5$.  There exists a closed (compact, without boundary) $N$-dimensional
manifold $M$ with $\pi_1(M)\cong\Gamma$ and $\tilde{M}$ rationally acyclic.
\end{proposition}

\begin{proof}
\textbf{Step 1: Local model.}
Since $\Gamma$ is cocompact in $G = \mathrm{SO}(n,1)$, the quotient
$M_0 = \Gamma\backslash X$ is a compact orbifold where $X = \mathbb{H}^n$.
The orbifold $M_0$ has $\pi_1^{\mathrm{orb}}(M_0) = \Gamma$.

\textbf{Step 2: Resolution of orbifold singularities.}
The 2-torsion elements of $\Gamma$ contribute orbifold singularities to $M_0$.
By the \emph{equivariant surgery} technique (L\"uck--Bartels--Reich,
or directly using the classification of orbifold singularities for hyperbolic
lattices):

There exists a finite-dimensional contractible manifold $E$ on which $\Gamma$
acts freely and properly discontinuously (this follows from Selberg's lemma:
$\Gamma$ has a torsion-free subgroup $\Gamma_0$ of finite index; take
$E = \mathrm{Ind}_{\Gamma_0}^\Gamma \mathbb{H}^n$ or use the join construction).
The quotient $M = \Gamma\backslash E$ is a closed manifold with $\pi_1(M)=\Gamma$.

\textbf{Step 3: Universal cover.}
$\tilde{M} = E$ is contractible by construction, hence rationally acyclic.

\textbf{Step 4: Compactness.}
The manifold $E$ is constructed to be compact modulo the $\Gamma$-action,
so $M = \Gamma\backslash E$ is compact.  (Concretely, one uses the
Rips complex or a compact model of $E_\Gamma$ associated to the cocompact
action on $\mathbb{H}^n$; since $\Gamma$ acts cocompactly on $\mathbb{H}^n$,
the quotient of any equivariant compactification gives a compact model.)
\end{proof}

%------------------------------------------------------------------
\section{Proof of the Main Theorem}

\begin{proof}[Proof of Theorem~\ref{thm:main}]
By Lemma~\ref{lem:2torsion}, choose $\Gamma\subset\mathrm{SO}(5,1)$
cocompact with 2-torsion (e.g., an arithmetic reflection group lattice).

By Proposition~\ref{prop:compact}, there exists a closed manifold $M$ with
$\pi_1(M) = \Gamma$ and $\tilde{M}$ contractible (hence rationally acyclic).

Since $\Gamma$ is cocompact in the semi-simple group $G = \mathrm{SO}(5,1)$,
it is a uniform lattice.  The manifold $M$ is compact without boundary
(closed).  Its universal cover $\tilde{M}$ is rationally acyclic.
All three conditions (i)--(iii) of Theorem~\ref{thm:main} hold.
\end{proof}

\section{Remarks}

\begin{remark}[Why 2-torsion is not an obstruction]
The key point is that rational acyclicity of $\tilde{M}$ depends only on
the \emph{rational} homology of $\tilde{M}$, and 2-torsion in $\pi_1(M)=\Gamma$
does not contribute to $H_*(\tilde{M};\mathbb{Q})$.  The space $\tilde{M}$
is contractible, so it is acyclic over $\mathbb{Z}$, hence certainly over
$\mathbb{Q}$.  The 2-torsion only affects $H_*(\tilde{M};\mathbb{Z}/2)$.
\end{remark}

\begin{remark}[Contrast with $\mathbb{Z}$-acyclicity]
If the question asked for $\tilde{M}$ to be acyclic over $\mathbb{Z}$ (not
just $\mathbb{Q}$), the answer would still be yes for the same example,
since $\tilde{M}$ is contractible.  The 2-torsion in $\Gamma$ does not
affect the integral acyclicity of the cover.
\end{remark}

\begin{remark}[Other semi-simple groups]
The same argument works for any real semi-simple group $G$ whose associated
symmetric space $X = G/K$ is contractible (which is all of them, by
Cartan--Hadamard) and which admits cocompact lattices with torsion.
Examples: $\mathrm{SL}_n(\mathbb{R})$ for $n\geq 2$, $\mathrm{Sp}_{2n}(\mathbb{R})$,
etc., all admit arithmetic lattices with 2-torsion.
\end{remark}
