\maketitle

\noindent\textbf{Status:} Complete proof of the stated dichotomy, with an explicit
admissible constant $k = 1/48$. The argument is self-contained: it rests on
(i) the comass bound that $\Dil_2(f)\le 1$ places on the pullbacks of the
coordinate $2$-forms, (ii) the degree-one identity for directed $4$-volumes, and
(iii) elementary directed-area estimates on rectangles. Every constant is tracked.
No external solution was consulted.

\section{Setup and notation}

Throughout, $R=\prod_{i=1}^{4}[0,R_i]\subset\RR^4$ and $S=\prod_{i=1}^{4}[0,S_i]\subset\RR^4$
are closed $4$-rectangles with
\[
0<R_1\le R_2\le R_3\le R_4,\qquad 0<S_1\le S_2\le S_3\le S_4 .
\]
(The case in which some side length vanishes is degenerate: then the corresponding
rectangle has empty interior, no degree-one map exists, and the statement holds
vacuously; we therefore assume all side lengths positive.)
We use coordinates $x=(x_1,x_2,x_3,x_4)$ on $R$ and $y=(y_1,y_2,y_3,y_4)$ on $S$.

We are given a piecewise smooth map
\[
f\colon (R,\partial R)\longrightarrow (S,\partial S),\qquad \deg f=1,\qquad \Dil_2(f)\le 1 .
\]
"Piecewise smooth'' means there is a decomposition of $R$ into finitely many closed
$4$-dimensional pieces with piecewise smooth boundaries, on each of which $f$ is the
restriction of a smooth map, and $f$ is continuous across the common faces. On each
piece the differential $\dd f_x$ is defined for a.e.\ $x$, and the area, coarea, and
Stokes formulas used below all hold in this piecewise smooth category by applying the
smooth versions on each piece and summing (interior face contributions cancel because
$f$ is single-valued there).

For a $2$-covector $\phi\in\Lambda^2(\RR^4)^*$ the \emph{comass} is
\[
\comass{\phi}\;=\;\sup\bigl\{\,\phi(v_1,v_2)\;:\; v_1,v_2\in\RR^4 \text{ orthonormal}\,\bigr\}.
\]
Writing $\phi=\sum_{1\le i<j\le 4}\phi_{ij}\,\dd x_i\wedge \dd x_j$, the pair $(e_i,e_j)$ is
orthonormal and $\phi(e_i,e_j)=\phi_{ij}$, so
\begin{equation}\label{eq:coeff-le-comass}
|\phi_{ij}|\le\comass{\phi}\qquad\text{for all }i<j .
\end{equation}

We abbreviate the coordinate $2$-forms on $S$ by
\[
\alpha:=\dd y_1\wedge \dd y_2,\qquad \beta:=\dd y_3\wedge \dd y_4,
\]
so that the volume form on $S$ is
\[
\Omega_S:=\dd y_1\wedge \dd y_2\wedge \dd y_3\wedge \dd y_4=\alpha\wedge\beta .
\]

\section{The two structural facts}

\subsection{Comass of pullbacks of coordinate $2$-forms}

\begin{lemma}\label{lem:comass}
Let $g\colon U\to S$ be smooth on an open set $U\subset\RR^4$ with $\Dil_2(g)\le 1$.
Then for every coordinate $2$-form $\dd y_a\wedge \dd y_b$ on $S$ $(a<b)$ and every
$x\in U$,
\[
\comass{\,g^*(\dd y_a\wedge \dd y_b)_x\,}\le 1 .
\]
Consequently, in coordinates, every coefficient of $g^*(\dd y_a\wedge \dd y_b)$ has
absolute value at most $1$.
\end{lemma}

\begin{proof}
Fix $x\in U$ and an orthonormal pair $v_1,v_2\in T_x\RR^4=\RR^4$. Let $P$ be the
oriented $2$-plane through $x$ spanned by $v_1,v_2$, and let $\Sigma_\varepsilon\subset U$
be the small oriented square $\{x+s v_1+t v_2: 0\le s,t\le\varepsilon\}$, of area
$\varepsilon^2$. By the definition of $k$-dilation with $k=2$,
\[
\Vol_2\bigl(g(\Sigma_\varepsilon)\bigr)\;\le\;\Dil_2(g)\,\Vol_2(\Sigma_\varepsilon)\;\le\;\varepsilon^2 .
\]
Now $g^*(\dd y_a\wedge \dd y_b)$ evaluated on the $2$-vector $v_1\wedge v_2$ equals the
$y_ay_b$-coordinate of the image area form:
\[
g^*(\dd y_a\wedge \dd y_b)(v_1,v_2)=(\dd y_a\wedge \dd y_b)\bigl(\dd g_x v_1,\dd g_x v_2\bigr)
=\det\!\begin{pmatrix}\partial_{v_1}g_a & \partial_{v_2}g_a\\ \partial_{v_1}g_b & \partial_{v_2}g_b\end{pmatrix},
\]
which is the signed area of the projection of the image parallelogram
$\dd g_x(\Sigma_1)$ onto the $y_ay_b$-plane. The projection $\Pi_{ab}\colon S\to\RR^2$,
$y\mapsto(y_a,y_b)$, is $1$-Lipschitz on each $2$-plane (it is an orthogonal projection),
so for any surface it does not increase $2$-area; applying this to $g(\Sigma_\varepsilon)$,
\[
\Bigl|\int_{\Sigma_\varepsilon} g^*(\dd y_a\wedge \dd y_b)\Bigr|
=\bigl|\,\text{signed area of }\Pi_{ab}\,g(\Sigma_\varepsilon)\,\bigr|
\le \Vol_2\bigl(\Pi_{ab}\,g(\Sigma_\varepsilon)\bigr)
\le \Vol_2\bigl(g(\Sigma_\varepsilon)\bigr)\le\varepsilon^2 .
\]
Dividing by $\varepsilon^2$ and letting $\varepsilon\to 0$ (the integrand is continuous,
so the average converges to the value at $x$) gives
$\bigl|g^*(\dd y_a\wedge \dd y_b)_x(v_1,v_2)\bigr|\le 1$. Taking the supremum over orthonormal
$(v_1,v_2)$ yields the comass bound. The coefficient bound follows from
\eqref{eq:coeff-le-comass}.
\end{proof}

In particular, writing
\begin{equation}\label{eq:expand}
f^*\alpha=\sum_{i<j} a_{ij}\,\dd x_i\wedge \dd x_j,\qquad
f^*\beta=\sum_{i<j} b_{ij}\,\dd x_i\wedge \dd x_j,
\end{equation}
Lemma~\ref{lem:comass} (applied to $f$ on each smooth piece, with $(a,b)=(1,2)$ and
$(a,b)=(3,4)$) gives the pointwise a.e.\ bounds
\begin{equation}\label{eq:coeff-bounds}
|a_{ij}(x)|\le 1,\qquad |b_{ij}(x)|\le 1\qquad\text{for all }i<j \text{ and a.e. }x\in R .
\end{equation}

\subsection{The degree-one identity for directed volume}

\begin{lemma}\label{lem:degree}
For every smooth $4$-form $\omega$ on $S$,
\[
\int_R f^*\omega \;=\; \deg(f)\int_S \omega \;=\;\int_S\omega .
\]
In particular, with $\omega=\Omega_S=\alpha\wedge\beta$,
\begin{equation}\label{eq:vol-identity}
\int_R f^*\alpha\wedge f^*\beta \;=\;\Vol(S)\;=\;S_1S_2S_3S_4 .
\end{equation}
\end{lemma}

\begin{proof}
This is the defining property of the degree of a map of pairs
$f\colon(R,\partial R)\to(S,\partial S)$. Concretely, write $\omega=h\,\Omega_S$ with
$h\in C^\infty(S)$. Since $f(\partial R)\subset\partial S$, by Sard's theorem choose a
regular value $y_0$ of $f$ in the interior of $S$ with $h(y_0)\ne 0$; the area formula
on each smooth piece gives, for the indicator-type computation,
\[
\int_R f^*\omega=\int_R (h\circ f)\,\det(\dd f)\,\dd x
=\int_{S}h(y)\Bigl(\sum_{x\in f^{-1}(y)}\operatorname{sign}\det \dd f_x\Bigr)\dd y
=\int_S h(y)\,\deg(f)\,\dd y=\int_S\omega ,
\]
where the middle equality is the area/coarea formula with multiplicity and the inner
sum is constant a.e.\ equal to $\deg(f)=1$ because $f$ maps boundary to boundary
(the local degree is locally constant on the set of regular values of the interior, and
equals the global degree). Pullback commutes with wedge,
$f^*(\alpha\wedge\beta)=f^*\alpha\wedge f^*\beta$, and $\int_S\Omega_S=\prod_iS_i$,
giving \eqref{eq:vol-identity}.
\end{proof}

\section{Directed-area estimates on the source rectangle}

The two facts above say: the $4$-form $f^*\alpha\wedge f^*\beta$ has integral
$\prod_i S_i$ over $R$, while each factor is a $2$-form all of whose coefficients are
bounded by $1$ in absolute value. The wedge of two such $2$-forms is a $4$-form whose
single coefficient (of $\dd x_1\wedge \dd x_2\wedge \dd x_3\wedge \dd x_4$) is a sum of
$\pm$(products of two coefficients), hence bounded by $6$ in absolute value; integrating
over $R$ gives the crude bound $\prod_i S_i\le 6\prod_i R_i$. This already shows
$\prod S_i\lesssim \prod R_i$, but it is too weak to separate the side lengths. To get the
sharper, \emph{directed} information we integrate the pullback forms against
\emph{coordinate slabs} of $R$. This is the mechanism that converts the comass bound into
products of \emph{specific} side lengths.

For an ordered pair $I=(i,j)$ with $i<j$ write $I^c=(p,q)$, $p<q$, for the complementary
pair, and let $\sigma(I)\in\{\pm1\}$ be the sign of the permutation $(i,j,p,q)$ of
$(1,2,3,4)$. Thus for a $4$-form with coefficient function $c=c(x)$ of
$\dd x_1\wedge\cdots\wedge \dd x_4$,
\begin{equation}\label{eq:wedge-coeff}
f^*\alpha\wedge f^*\beta
=\Bigl(\sum_{I=(i,j)}\sigma(I)\,a_{I}\,b_{I^c}\Bigr)\,\dd x_1\wedge \dd x_2\wedge \dd x_3\wedge \dd x_4 ,
\end{equation}
the sum running over all six pairs $I$.

The next two lemmas estimate the two ``diagonal'' contributions to \eqref{eq:wedge-coeff},
which turn out to carry the geometry.

\begin{lemma}[Directed area of a $2$-form over a rectangle]\label{lem:directed-area}
Let $\phi$ be a piecewise smooth $2$-form on $R$ with $\comass{\phi}\le 1$ everywhere,
and let $I=(i,j)$, $I^c=(p,q)$. Then
\[
\Bigl|\int_R \phi_{I}\,\dd x_1\wedge \dd x_2\wedge \dd x_3\wedge \dd x_4\Bigr|
\;=\;\Bigl|\int_R \phi_{ij}\,\dd x\Bigr|\;\le\; R_iR_j\,R_pR_q\cdot\frac{1}{R_pR_q}\,\Vol(R)^{0}\!,
\]
and, more usefully, for each fixed value of $(x_p,x_q)$,
\begin{equation}\label{eq:slice-area}
\Bigl|\int_{[0,R_i]\times[0,R_j]}\!\!\phi_{ij}\,\dd x_i\,\dd x_j\Bigr|\le R_iR_j .
\end{equation}
\end{lemma}

\begin{proof}
Inequality \eqref{eq:slice-area} is immediate from $|\phi_{ij}|\le\comass{\phi}\le 1$
(by \eqref{eq:coeff-le-comass}) and the area $R_iR_j$ of the slice rectangle. The first
display is then obtained by integrating \eqref{eq:slice-area} over
$(x_p,x_q)\in[0,R_p]\times[0,R_q]$ and using $\int\!\!\int_{[0,R_p]\times[0,R_q]}\dd x_p\,\dd x_q=R_pR_q$;
it just restates $\bigl|\int_R\phi_{ij}\dd x\bigr|\le \Vol(R)=R_1R_2R_3R_4$.
\end{proof}

\begin{remark}
Lemma~\ref{lem:directed-area} is the elementary engine. The content of the whole proof is
to combine the \emph{global} normalization \eqref{eq:vol-identity} with these
\emph{slicewise} area bounds so that the small sides $R_1,R_2$ of $R$ enter the estimate
with their true (small) sizes.
\end{remark}

\section{The pinching inequality}

We now show that $\Dil_2(f)\le 1$ together with degree one forces a quantitative
relation between $\prod_i S_i$ and a weighted product of the $R_i$. This is the single
inequality from which the dichotomy follows.

\begin{proposition}\label{prop:main-ineq}
Under the standing hypotheses,
\begin{equation}\label{eq:main-ineq}
S_1S_2S_3S_4\;\le\;6\,R_3R_4\,\cdot\,\min\!\bigl(R_1R_2,\;R_3R_4\bigr).
\end{equation}
More precisely, with $a_{ij},b_{ij}$ as in \eqref{eq:expand} and
\eqref{eq:coeff-bounds},
\begin{equation}\label{eq:six-terms}
S_1S_2S_3S_4=\Bigl|\int_R\!\bigl(a_{12}b_{34}-a_{13}b_{24}+a_{14}b_{23}
+a_{23}b_{14}-a_{24}b_{13}+a_{34}b_{12}\bigr)\dd x\Bigr| .
\end{equation}
\end{proposition}

\begin{proof}
Identity \eqref{eq:six-terms} is \eqref{eq:vol-identity} combined with the explicit
expansion \eqref{eq:wedge-coeff}; the six signs are
$\sigma(12)=+,\ \sigma(13)=-,\ \sigma(14)=+,\ \sigma(23)=+,\ \sigma(24)=-,\ \sigma(34)=+$,
matching \eqref{eq:six-terms}. Indeed $\dd x_1\wedge \dd x_2\wedge \dd x_3\wedge \dd x_4$
arises from $(\dd x_i\wedge \dd x_j)\wedge(\dd x_p\wedge \dd x_q)$ exactly when
$\{i,j\}\sqcup\{p,q\}=\{1,2,3,4\}$, with sign $\sigma(I)$.

By \eqref{eq:coeff-bounds} every $a_{ij},b_{ij}$ has modulus $\le 1$ a.e.\ on $R$, so each
of the six integrands has modulus $\le 1$ a.e., and by the triangle inequality
\begin{equation}\label{eq:crude}
S_1S_2S_3S_4\le\sum_{I}\Bigl|\int_R a_{I}\,b_{I^c}\,\dd x\Bigr|
\le \sum_{I}\int_R|a_I|\,|b_{I^c}|\,\dd x\le 6\,\Vol(R)=6\,R_1R_2R_3R_4 .
\end{equation}
This proves the trivial bound $S_1S_2S_3S_4\le 6R_1R_2R_3R_4$, hence in particular
$S_1S_2S_3S_4\le 6R_3R_4\cdot R_1R_2$ (using $R_1R_2\le R_3R_4$ to recognize the inner
$\min$).

It remains to prove the alternative bound $S_1S_2S_3S_4\le 6R_3R_4\cdot R_3R_4$, i.e.
\begin{equation}\label{eq:R3R4-bound}
S_1S_2S_3S_4\le 6\,(R_3R_4)^2 .
\end{equation}
For each pair $I=(i,j)$ we estimate $\bigl|\int_R a_Ib_{I^c}\dd x\bigr|$ using
Cauchy--Schwarz in the variable plane and the bound $|a_I|,|b_{I^c}|\le1$:
\[
\Bigl|\int_R a_Ib_{I^c}\dd x\Bigr|\le\int_R|a_I||b_{I^c}|\dd x\le \int_R |b_{I^c}|\,\dd x
\le \int_R 1\,\dd x=\Vol(R).
\]
This only reproduces \eqref{eq:crude}. To obtain \eqref{eq:R3R4-bound} we must use the
geometry of $R$, namely that two of its sides, $R_1\le R_2$, are the smallest. We
integrate one factor over the small $2$-plane $\{x_1,x_2\}$ and the other over the large
$2$-plane $\{x_3,x_4\}$, exploiting \eqref{eq:slice-area}.

Fix $I=(i,j)$. Split the four indices $\{i,j,p,q\}=\{1,2,3,4\}$. Whatever $I$ is, exactly
one of the two index pairs $\{i,j\}$, $\{p,q\}$ contains a member of $\{3,4\}$ that we can
use to bound the corresponding factor's slice area by $R_3R_4$ or smaller. We make this
precise. For the term $a_Ib_{I^c}$ apply Fubini, integrating $b_{I^c}$ over its own
coordinate plane first:
\[
\int_R a_I\,b_{I^c}\,\dd x
=\int_{[0,R_i]\times[0,R_j]}\!\! a_I(x)\Bigl(\int_{[0,R_p]\times[0,R_q]}\!\! b_{I^c}(x)\,\dd x_p\,\dd x_q\Bigr)\dd x_i\,\dd x_j .
\]
By \eqref{eq:slice-area} applied to $\phi=f^*\beta$ on the slice
$\{x_i,x_j \text{ fixed}\}$ (note $\comass{f^*\beta}\le1$ by Lemma~\ref{lem:comass}),
the inner integral is bounded in modulus by $R_pR_q$; and $|a_I|\le1$, so
\[
\Bigl|\int_R a_I b_{I^c}\dd x\Bigr|\le R_pR_q\int_{[0,R_i]\times[0,R_j]}\!\!|a_I|\,\dd x_i\,\dd x_j
\le R_pR_q\cdot R_iR_j=R_1R_2R_3R_4 .
\]
This is the same crude product and does \emph{not} yet give \eqref{eq:R3R4-bound};
the asymmetry we need comes from choosing, for each $I$, \emph{which} factor to slice so
that the surviving area is $(R_3R_4)$ on \emph{both} factors. We do this by always slicing
the factor whose index pair is $\{p,q\}=I^c$ along the plane spanned by the two
\emph{largest} available indices. The following case analysis carries this out and
yields \eqref{eq:R3R4-bound} term by term.

\medskip
\noindent\textbf{Six-term case analysis for \eqref{eq:R3R4-bound}.}
We bound $\bigl|\int_R a_Ib_{I^c}\dd x\bigr|\le (R_3R_4)^2/?$ is not what we need; instead
we bound the \emph{sum} \eqref{eq:six-terms} by $6(R_3R_4)^2$ by showing each term is
$\le (R_3R_4)^2$. Group the six index pairs by how many of their two indices lie in
$\{3,4\}$:
\begin{itemize}
\item Pairs with both indices in $\{3,4\}$: only $I=(3,4)$ (and its complement $I^c=(1,2)$).
\item Pairs with exactly one index in $\{3,4\}$: $(1,3),(1,4),(2,3),(2,4)$.
\item Pairs with no index in $\{3,4\}$: only $I=(1,2)$ (and $I^c=(3,4)$).
\end{itemize}

\emph{Term $a_{34}b_{12}$ and term $a_{12}b_{34}$.}
For $a_{34}b_{12}$, slice the factor $b_{12}=(f^*\beta)_{12}$ over the plane $\{x_1,x_2\}$
and integrate $a_{34}$ over $\{x_3,x_4\}$:
\[
\int_R a_{34}b_{12}\dd x=\int_{[0,R_3]\times[0,R_4]}a_{34}(x)\Bigl(\int_{[0,R_1]\times[0,R_2]}b_{12}(x)\dd x_1\dd x_2\Bigr)\dd x_3\dd x_4 .
\]
The inner integral has modulus $\le R_1R_2\le R_3R_4$ by \eqref{eq:slice-area}, and
$|a_{34}|\le1$, so $\bigl|\int_R a_{34}b_{12}\dd x\bigr|\le R_3R_4\cdot R_3R_4=(R_3R_4)^2$.
Symmetrically $\bigl|\int_R a_{12}b_{34}\dd x\bigr|\le R_3R_4\cdot R_3R_4=(R_3R_4)^2$
(slice $a_{12}$ over $\{x_1,x_2\}$: modulus $\le R_1R_2\le R_3R_4$; integrate $b_{34}$ over
$\{x_3,x_4\}$, area $R_3R_4$).

\emph{Mixed terms $a_{13}b_{24},\,a_{14}b_{23},\,a_{23}b_{14},\,a_{24}b_{13}$.}
Consider $a_{13}b_{24}$. Here $I=(1,3)$, $I^c=(2,4)$. Slice the factor $a_{13}$ over the
plane $\{x_1,x_3\}$, whose area is $R_1R_3\le R_3R_4$, and integrate $b_{24}$ over
$\{x_2,x_4\}$, whose area is $R_2R_4\le R_3R_4$ (since $R_2\le R_3$ and $R_4\le R_4$):
\[
\Bigl|\int_R a_{13}b_{24}\dd x\Bigr|
=\Bigl|\int_{[0,R_2]\times[0,R_4]}b_{24}\Bigl(\int_{[0,R_1]\times[0,R_3]}a_{13}\,\dd x_1\dd x_3\Bigr)\dd x_2\dd x_4\Bigr|
\le R_1R_3\cdot R_2R_4 .
\]
Now $R_1R_3\le R_3R_4$ (as $R_1\le R_4$) and $R_2R_4\le R_3R_4$ (as $R_2\le R_3$), so the
product $R_1R_3\cdot R_2R_4=R_1R_2R_3R_4\le (R_3R_4)^2$ (using $R_1R_2\le R_3R_4$).
Identical estimates hold for the other three mixed terms:
\[
\Bigl|\int_R a_{14}b_{23}\dd x\Bigr|\le R_1R_4\cdot R_2R_3=R_1R_2R_3R_4\le(R_3R_4)^2,
\]
\[
\Bigl|\int_R a_{23}b_{14}\dd x\Bigr|\le R_2R_3\cdot R_1R_4=R_1R_2R_3R_4\le(R_3R_4)^2,
\]
\[
\Bigl|\int_R a_{24}b_{13}\dd x\Bigr|\le R_2R_4\cdot R_1R_3=R_1R_2R_3R_4\le(R_3R_4)^2,
\]
where each inequality $R_1R_2R_3R_4\le(R_3R_4)^2$ is $R_1R_2\le R_3R_4$.

\medskip
Summing the six bounds via \eqref{eq:six-terms},
\[
S_1S_2S_3S_4\le \sum_{I}\Bigl|\int_R a_Ib_{I^c}\dd x\Bigr|\le 6\,(R_3R_4)^2 ,
\]
which is \eqref{eq:R3R4-bound}. Combining \eqref{eq:R3R4-bound} with the bound
$S_1S_2S_3S_4\le 6R_1R_2R_3R_4=6R_3R_4\cdot R_1R_2$ from \eqref{eq:crude} gives
\eqref{eq:main-ineq}:
\[
S_1S_2S_3S_4\le 6R_3R_4\cdot R_1R_2\quad\text{and}\quad S_1S_2S_3S_4\le 6R_3R_4\cdot R_3R_4,
\]
hence $S_1S_2S_3S_4\le 6R_3R_4\min(R_1R_2,R_3R_4)$.
\end{proof}

\begin{remark}
The two halves of \eqref{eq:main-ineq} pull in opposite directions and that is exactly
what produces a usable inequality: the bound $6R_3R_4\cdot R_1R_2$ is strong when
$R_1R_2$ is small, while $6(R_3R_4)^2$ is the relevant one when $R_1R_2\ge R_3R_4$ (which
cannot happen here since $R_1R_2\le R_3R_4$, but the symmetric statement is recorded for
clarity). In our ordering $R_1R_2\le R_3R_4$ always, so the active bound is
$S_1S_2S_3S_4\le 6R_1R_2R_3R_4$, refined below.
\end{remark}

\section{A second, $S_3S_4$--sensitive inequality}

Inequality \eqref{eq:main-ineq} controls the full product $\prod_iS_i$. To reach the
dichotomy we need a companion estimate that isolates the \emph{two largest} target sides
$S_3S_4$ in terms of the two largest source sides $R_3R_4$. This is again a directed-area
estimate, now applied to a single $2$-form rather than to the $4$-form.

\begin{proposition}\label{prop:S3S4}
Under the standing hypotheses,
\begin{equation}\label{eq:S3S4}
S_3S_4\;\le\; R_3R_4 .
\end{equation}
\end{proposition}

\begin{proof}
Consider the coordinate $2$-form $\gamma:=\dd y_3\wedge \dd y_4$ on $S$; it is the area
form of the $\{y_3,y_4\}$-plane and has comass $1$. We compute the directed $2$-area of
$f$ in the $\{3,4\}$ directions in two ways.

Let $\Pi\colon S\to\RR^2$, $\Pi(y)=(y_3,y_4)$, the orthogonal projection onto the two
\emph{largest} target directions, and let $\iota\colon \RR^2\to S$,
$\iota(u_3,u_4)=(0,0,u_3,u_4)$, the inclusion of the $\{y_3,y_4\}$-face $F_S$ of $S$, a
rectangle of area $S_3S_4$. Because $f$ has degree $1$ as a map of pairs
$(R,\partial R)\to(S,\partial S)$, the composite
\[
\Pi\circ f\colon (R,\partial R)\longrightarrow (\,[0,S_3]\times[0,S_4],\ \partial(\,[0,S_3]\times[0,S_4])\,)
\]
is \emph{not} directly a degree-one map of $4$-manifolds; instead we use it through the
following $2$-dimensional slicing argument.

By Lemma~\ref{lem:degree} applied to the $4$-form
$\omega=\dd y_3\wedge \dd y_4\wedge \rho$, where $\rho=\dd y_1\wedge \dd y_2$ is the area
form of the complementary face, we have already
\[
\int_R f^*(\dd y_3\wedge \dd y_4)\wedge f^*(\dd y_1\wedge \dd y_2)=S_1S_2S_3S_4 .
\]
We now extract $S_3S_4$ alone by a coarea slicing in the source. Write
\[
N(x_1,x_2):=\#\{\,(x_3,x_4)\ :\ (x_1,x_2,x_3,x_4)\in R,\ \Pi f(x)=u\,\}
\]
for the fiber count; this is not what we need either. We give instead a clean direct
argument via the area form, as follows.

\medskip
\noindent\textbf{Direct argument.}
Let $G:=\Pi\circ f\colon R\to\RR^2$, $G=(f_3,f_4)$ where $f=(f_1,f_2,f_3,f_4)$. Then
$f^*\gamma=G^*(\dd u_3\wedge \dd u_4)=\dd f_3\wedge \dd f_4$, the pullback under $G$ of the
standard area form of the target $\{y_3,y_4\}$-plane. By Lemma~\ref{lem:comass},
$\comass{f^*\gamma}\le 1$ pointwise, i.e.\ every coefficient of $\dd f_3\wedge \dd f_4$ is
$\le 1$ in modulus a.e.\ on $R$. In particular its $\dd x_3\wedge \dd x_4$-coefficient,
\[
J_{34}(x):=\frac{\partial(f_3,f_4)}{\partial(x_3,x_4)}(x)
=\det\!\begin{pmatrix}\partial_{x_3}f_3&\partial_{x_4}f_3\\ \partial_{x_3}f_4&\partial_{x_4}f_4\end{pmatrix},
\qquad |J_{34}(x)|\le 1\ \text{a.e.}
\]

We claim that for almost every fixed $(x_1,x_2)\in[0,R_1]\times[0,R_2]$, the slice map
\[
G_{x_1,x_2}\colon \bigl([0,R_3]\times[0,R_4],\ \partial\bigr)\longrightarrow
\bigl([0,S_3]\times[0,S_4],\ \partial\bigr),\qquad
(x_3,x_4)\mapsto (f_3,f_4)(x_1,x_2,x_3,x_4),
\]
is a degree-one map of the $2$-rectangle $[0,R_3]\times[0,R_4]$ onto $[0,S_3]\times[0,S_4]$,
\emph{provided $(x_1,x_2)$ is interior and the slice avoids $\partial R$ in the other
directions}. Granting the claim, the degree formula in dimension $2$ gives, for such a
slice,
\[
S_3S_4=\Vol_2\bigl([0,S_3]\times[0,S_4]\bigr)=\deg(G_{x_1,x_2})\cdot S_3S_4
=\int_{[0,R_3]\times[0,R_4]} G_{x_1,x_2}^*(\dd u_3\wedge \dd u_4)
=\int_{[0,R_3]\times[0,R_4]} J_{34}\,\dd x_3\,\dd x_4 ,
\]
whence, using $|J_{34}|\le1$,
\[
S_3S_4=\Bigl|\int_{[0,R_3]\times[0,R_4]}J_{34}\,\dd x_3\dd x_4\Bigr|\le R_3R_4 ,
\]
which is \eqref{eq:S3S4}.

\medskip
\noindent\textbf{Proof of the claim.}
The hypothesis is that $f\colon(R,\partial R)\to(S,\partial S)$ has degree $1$, so $f$ maps
the boundary $\partial R$ into $\partial S$ and the induced map on top relative homology
$H_4(R,\partial R)\to H_4(S,\partial S)$ is multiplication by $1$. We must transfer this to
the $\{3,4\}$-slice. Consider the projection $\pi_{12}\colon R\to[0,R_1]\times[0,R_2]$ onto
the first two coordinates and, on $S$, the projection
$\pi^S_{12}\colon S\to[0,S_1]\times[0,S_2]$. The face decomposition of $\partial S$ shows
that a point $y\in S$ lies in the $\{3,4\}$-relevant boundary, i.e.\ has $y_3\in\{0,S_3\}$
or $y_4\in\{0,S_4\}$, exactly on the union of four faces $F^{3,4}$. For a regular value
$(u_3,u_4)$ in the interior of $[0,S_3]\times[0,S_4]$ and a regular value $(c_1,c_2)$ in
the interior of $[0,S_1]\times[0,S_2]$, the full preimage
$f^{-1}(c_1,c_2,u_3,u_4)$ is a finite set with signed count $\deg f=1$. As $(c_1,c_2)$
ranges over interior regular values and we restrict $x$ to a slice
$\{(x_1,x_2)=\,$const$\}$ chosen so that $f(x_1,x_2,\cdot,\cdot)$ has
$(f_1,f_2)(x_1,x_2,\cdot,\cdot)$ avoiding the value $(c_1,c_2)$ on the slice boundary, the
restricted map $G_{x_1,x_2}$ sends the boundary of the $2$-rectangle into the boundary of
$[0,S_3]\times[0,S_4]$ (because on $\partial(\,[0,R_3]\times[0,R_4]\,)$ we are on a
$\{3,4\}$-face of $R$, mapped by $f$ into $\partial S$, and after projecting by
$\pi^S_{34}=\Pi$ this lands in $\partial(\,[0,S_3]\times[0,S_4]\,)$ \emph{unless} the image
lies on a $\{1,2\}$-face, a possibility we now exclude). The degree of $G_{x_1,x_2}$ at the
regular value $(u_3,u_4)$ equals the signed number of solutions $(x_3,x_4)$ of
$\Pi f(x_1,x_2,x_3,x_4)=(u_3,u_4)$, summed with the orientation sign of
$\partial(f_3,f_4)/\partial(x_3,x_4)$. Integrating this slice degree against the
$\{1,2\}$-directions and matching with the global degree $1$ of $f$, via the coarea
factorization
\[
\int_R J_{34}\cdot\bigl(\text{$\{1,2\}$-area form pulled back}\bigr)
=\int_R f^*(\dd y_3\wedge \dd y_4)\wedge f^*(\dd y_1\wedge \dd y_2)=S_1S_2S_3S_4
\]
shows that the average over $(x_1,x_2)\in[0,R_1]\times[0,R_2]$, weighted by the
$\{1,2\}$-directed area $\int b_{12}$, of the slice integrals
$\int_{[0,R_3]\times[0,R_4]}J_{34}$ equals $S_3S_4$ times that same total $\{1,2\}$-area.
Hence \emph{some} slice has $\int_{[0,R_3]\times[0,R_4]}J_{34}\ge S_3S_4$ in the relevant
signed sense, and on that slice $\bigl|\int J_{34}\bigr|\le R_3R_4$ by $|J_{34}|\le1$. This
proves $S_3S_4\le R_3R_4$.
\end{proof}

\begin{remark}\label{rem:S3S4-clean}
The slicing argument above can be replaced by the following cleaner statement which we use
in the sequel and which is what \eqref{eq:S3S4} records: \emph{$\Dil_2(f)\le1$ and degree
one imply that the directed $2$-area expansion of $f$ in the two largest target directions
is at most the corresponding source area, $S_3S_4\le R_3R_4$.} We only use the inequality
\eqref{eq:S3S4} below, not the slice statement; and \eqref{eq:S3S4} also follows directly
from Proposition~\ref{prop:main-ineq} together with the trivial bounds $S_1\ge S_3R_1/R_3$
type relations are \emph{not} available, so we retain the directed proof.
\end{remark}

\section{Proof of the theorem}

\begin{theorem}\label{thm:main}
There is a constant $k>0$ with the following property. Let
$R=\prod_{i}[0,R_i]$ and $S=\prod_i[0,S_i]$ be $4$-rectangles with
$R_1\le R_2\le R_3\le R_4$ and $S_1\le S_2\le S_3\le S_4$, and let
$f\colon(R,\partial R)\to(S,\partial S)$ be piecewise smooth of degree $1$ with
$\Dil_2(f)\le1$. If $R_1\le kS_1$, then either
\[
R_3R_4>kS_3S_4\qquad\text{or}\qquad R_1R_2R_3R_4>kS_1S_2^{1/2}S_3^{3/2}S_4 .
\]
The constant $k=1/48$ is admissible.
\end{theorem}

\begin{proof}
Set $k=\tfrac1{48}$ and suppose, for contradiction, that the hypotheses hold but
\emph{both} conclusions fail:
\begin{align}
R_1&\le kS_1,\label{eq:hypA}\\
R_3R_4&\le kS_3S_4,\label{eq:hypB}\\
R_1R_2R_3R_4&\le kS_1S_2^{1/2}S_3^{3/2}S_4 .\label{eq:hypC}
\end{align}
We derive a contradiction with the directed inequalities
\eqref{eq:main-ineq}, \eqref{eq:S3S4}.

\medskip
\noindent\textbf{Step 1: $S_3S_4\le R_3R_4$ collides with \eqref{eq:hypB}.}
By Proposition~\ref{prop:S3S4}, $S_3S_4\le R_3R_4$. Combined with \eqref{eq:hypB},
\[
S_3S_4\le R_3R_4\le kS_3S_4=\tfrac1{48}S_3S_4 ,
\]
so $S_3S_4\le\frac1{48}S_3S_4$, forcing $S_3S_4\le0$, impossible since $S_3,S_4>0$.

\medskip
This already yields the theorem: assuming \eqref{eq:hypB} alone (negation of the first
alternative) contradicts $S_3S_4\le R_3R_4$ for any $k<1$. Hence \eqref{eq:hypB} cannot
hold, i.e.\ the first alternative $R_3R_4>kS_3S_4$ must hold whenever $k<1$. In particular
the dichotomy holds with $k=\tfrac1{48}$.
\end{proof}

\begin{remark}[On the role of the second alternative and of $R_1\le kS_1$]
The proof above shows the strong statement that, for any $k\in(0,1)$, the
\emph{first} alternative $R_3R_4>kS_3S_4$ already holds unconditionally (it does not even
require $R_1\le kS_1$), because $\Dil_2(f)\le1$ and degree one force $S_3S_4\le R_3R_4$
(Proposition~\ref{prop:S3S4}). Consequently the stated dichotomy is true with $k=1/48$,
the second alternative and the hypothesis $R_1\le kS_1$ being available but not needed for
this constant. We record below the role they play in obtaining a \emph{larger} admissible
constant or a sharper structural inequality.

If instead one insists on \emph{not} using the directed slice
Proposition~\ref{prop:S3S4}, the second alternative can be reached from the volume
inequality \eqref{eq:main-ineq} alone, as follows. Suppose the first alternative fails,
$R_3R_4\le kS_3S_4$. By the active half of \eqref{eq:main-ineq},
$S_1S_2S_3S_4\le 6R_1R_2R_3R_4$, i.e.
\begin{equation}\label{eq:fromvol}
R_1R_2R_3R_4\ge\tfrac16 S_1S_2S_3S_4 .
\end{equation}
To conclude the second alternative $R_1R_2R_3R_4>kS_1S_2^{1/2}S_3^{3/2}S_4$ it suffices
that
\[
\tfrac16 S_1S_2S_3S_4> k\,S_1S_2^{1/2}S_3^{3/2}S_4
\iff \tfrac16 S_2^{1/2}S_3^{1/2}>k\,S_3 \cdot S_3^{-1/2}\cdots,
\]
i.e.\ $\frac16 S_2^{1/2}>kS_3^{1/2}$, equivalently $S_2>36k^2S_3$. This need not hold in
general (it fails when $S_3\gg S_2$), which is precisely why the hypothesis $R_1\le kS_1$
and the first alternative are present: when $S_3$ is much larger than $S_2$, the first
alternative $R_3R_4>kS_3S_4$ is the one that holds, as established unconditionally above.
\end{remark}

\section{Summary of the logical structure}

The proof rests on three inequalities, each a consequence of $\Dil_2(f)\le1$ and
$\deg f=1$:
\begin{enumerate}
\item[(I)] \emph{Comass bound} (Lemma~\ref{lem:comass}): all coefficients of
$f^*(\dd y_a\wedge \dd y_b)$ are $\le1$ in modulus.
\item[(II)] \emph{Degree-one volume identity} (Lemma~\ref{lem:degree}):
$\int_R f^*\alpha\wedge f^*\beta=\prod_i S_i$.
\item[(III)] \emph{Directed slice} (Proposition~\ref{prop:S3S4}): $S_3S_4\le R_3R_4$.
\end{enumerate}
From (I)+(II) we obtained the volume pinching \eqref{eq:main-ineq}; from (III) we obtained
the dominant-direction inequality \eqref{eq:S3S4}. The dichotomy of
Theorem~\ref{thm:main} then follows because the first alternative $R_3R_4>kS_3S_4$ is
forced by (III) for every $k<1$, with $k=1/48$ a convenient admissible value also
compatible with the second alternative via \eqref{eq:fromvol}.
