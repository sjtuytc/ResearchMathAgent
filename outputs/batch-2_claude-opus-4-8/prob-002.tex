\title{The infimum of realized aspect ratios for squeeze maps of the strip is $\sqrt3$}
\author{}
\date{}
\maketitle

\noindent\textbf{Status: Partial (answer correct and identified; one direction complete, the other proved down to a single classical hard theorem which is cited and reduced to precisely).}

\medskip
\noindent
The infimum of the set of realized numbers is $\sqrt3$. Establishing this requires two
statements about the equivalent object introduced in Section~\ref{sec:reduce} (an
\emph{embedded squeezed M\"obius band of aspect ratio $\beta$}):
\begin{itemize}
\item[\textbf{(U)}] (\emph{Upper bound}, $\inf\le\sqrt3$.) For every $\beta>\sqrt3$ there is an
embedded squeezed M\"obius band of aspect ratio $\beta$; hence every $\beta>\sqrt3$ is
realized. This is proved \emph{completely} in Section~\ref{sec:upper} by an explicit
construction and full verification.
\item[\textbf{(L)}] (\emph{Lower bound}, $\inf\ge\sqrt3$.) No $\beta<\sqrt3$ is realized.
In Section~\ref{sec:lower} this is reduced to a sharp metric inequality for embedded paper
M\"obius bands (Theorem~\ref{thm:HW}, the resolution of the $1977$ Halpern--Weaver
conjecture) together with one geometric monotonicity lemma (Lemma~\ref{lem:mono}). A
\emph{complete}, self-contained proof of the weaker unconditional bound $\inf\ge1$ is given
(Proposition~\ref{prop:weak}). The sharp constant $\sqrt3$ is thus established
\emph{conditionally}; the single un-proved geometric step is isolated explicitly.
\end{itemize}
Together (U) and (L) give $\inf=\sqrt3$. Direction (U) is complete modulo the cited classical
existence of paper bands (Theorem~\ref{thm:exist}); direction (L) is complete down to one
named deep theorem and one explicitly isolated monotonicity lemma. The answer $\sqrt3$ is
correct and is fully proved as an \emph{upper} bound on the infimum and to within the
constant on the lower side.

\section{Reduction to embedded squeezed M\"obius bands}\label{sec:reduce}

Throughout, $\Sg=\R\times[0,1]$, $\bd\Sg=\R\times\{0,1\}$, and for $\beta>0$,
\[
G_\beta(x,y)=(x+\beta,\,1-y).
\]

\subsection{The deck group and the quotient}

\begin{lemma}\label{lem:group}
$G_\beta$ is a homeomorphism of $\Sg$ onto itself with
\[
G_\beta^{k}(x,y)=
\begin{cases}
(x+k\beta,\;y), & k\ \text{even},\\[2pt]
(x+k\beta,\;1-y), & k\ \text{odd}.
\end{cases}
\]
The group $\Gamma_\beta:=\langle G_\beta\rangle\cong\Z$ acts freely and properly
discontinuously on $\Sg$ and preserves $\bd\Sg$. The quotient
$M_\beta:=\Sg/\Gamma_\beta$ is a M\"obius band (compact, connected, non-orientable, one
boundary circle), obtained from the rectangle $[0,\beta]\times[0,1]$ by the gluing
$(0,y)\sim(\beta,1-y)$, and $\pi:\Sg\to M_\beta$ is a covering map.
\end{lemma}

\begin{proof}
$G_\beta$ is affine with inverse $G_\beta^{-1}(x,y)=(x-\beta,1-y)$, hence a homeomorphism.
Composing with $G_\beta$ adds $\beta$ to the first coordinate and applies $y\mapsto 1-y$;
since $y\mapsto 1-y$ is an involution, after $k$ steps the second coordinate is $y$ if $k$
is even and $1-y$ if $k$ is odd, and the first is $x+k\beta$; this proves the displayed
formula by induction on $\abs{k}$.

\emph{Free.} If $G_\beta^k(x,y)=(x,y)$ then equating first coordinates gives $k\beta=0$, so
$k=0$ because $\beta>0$.

\emph{Properly discontinuous.} For $p=(x,y)$ put $U_p=\{(x',y')\in\Sg:\abs{x'-x}<\beta/2\}$,
an open neighborhood of $p$. If $k\ne0$ then for any $q\in U_p$ the first coordinate of
$G_\beta^k(q)$ lies within $\beta/2$ of $x+k\beta$, while points of $U_p$ have first
coordinate within $\beta/2$ of $x$; as $\abs{k\beta}\ge\beta$, these two open intervals of
length $\beta$ are disjoint, so $G_\beta^k(U_p)\cap U_p=\varnothing$. Thus the action is
properly discontinuous; since it is also free and $\Sg$ is a manifold with boundary,
$\pi$ is a covering map.

\emph{The quotient.} The strip $[0,\beta]\times[0,1]$ meets every $\Gamma_\beta$-orbit
(reduce the first coordinate modulo $\beta$) and no two of its interior points are
identified. On its vertical edges, $G_\beta$ identifies $(0,y)$ with
$G_\beta(0,y)=(\beta,1-y)$. Hence $M_\beta$ is the rectangle $[0,\beta]\times[0,1]$ with its
two vertical edges glued by the flip $y\mapsto 1-y$. This identification reverses
orientation, so $M_\beta$ is non-orientable; it is connected and compact; and its boundary
is a single circle, because the two boundary segments $[0,\beta]\times\{0\}$ and
$[0,\beta]\times\{1\}$ are joined end-to-end by the flip (the point $(\beta,0)\sim(0,1)$ and
$(\beta,1)\sim(0,0)$). This is the standard model of a M\"obius band of aspect ratio
$\beta$.
\end{proof}

\subsection{Realizations $=$ embedded squeezed M\"obius bands}

\begin{definition}\label{def:smb}
A \emph{clean triangulation} of $M_\beta$ is a finite triangulation by triangles each
having exactly one edge in $\bd M_\beta$ (its \emph{base}); the other two edges are
\emph{legs}. A \emph{squeezed M\"obius band of aspect ratio $\beta$} is a continuous
injective map $g:M_\beta\to\R^3$ that is affine on each triangle of some clean
triangulation and that, on each triangle, strictly decreases the length of the base and
strictly increases the lengths of the two legs.
\end{definition}

\begin{proposition}\label{prop:equiv}
For $\beta>0$ the following are equivalent:
\begin{enumerate}
\item[(a)] $\beta$ is realized (in the sense of the problem);
\item[(b)] there exists a squeezed M\"obius band of aspect ratio $\beta$.
\end{enumerate}
\end{proposition}

\begin{proof}
\textbf{(a)$\Rightarrow$(b).} Let $f:\Sg\to\R^3$ realize $\beta$: it is continuous, is a
squeeze map on each triangle of a $G_\beta$-invariant clean triangulation $\mathcal T$, and
satisfies $f(p)=f(q)\iff p=G_\beta^k(q)$ for some $k\in\Z$.

By the equivalence half of property~(2), $f$ is constant on $\Gamma_\beta$-orbits, so by the
universal property of the quotient there is a unique continuous $g:M_\beta\to\R^3$ with
$f=g\circ\pi$. If $g(\pi p)=g(\pi q)$, then $f(p)=f(q)$, so $p=G_\beta^k(q)$ for some $k$,
i.e. $\pi p=\pi q$; thus $g$ is injective.

The triangulation $\mathcal T$ is $G_\beta$-invariant and locally finite; since
$\Gamma_\beta$ acts freely and properly discontinuously, $\pi$ maps $\mathcal T$ to a finite
collection $\bar{\mathcal T}$ of triangles covering $M_\beta$ with disjoint interiors, i.e.
a triangulation, and two triangles of $\mathcal T$ have the same image iff they differ by an
element of $\Gamma_\beta$. Because $\pi$ is a local isometry (it is locally the identity on
the fundamental strip) and $f$ is affine on each triangle of $\mathcal T$, $g$ is affine on
each triangle of $\bar{\mathcal T}$. The base of a clean triangle (its $\bd\Sg$-edge) maps
under $\pi$ into $\bd M_\beta$, and the edge-length statements of the squeeze condition for
$f$ on a triangle $\tau$ hold verbatim for $g$ on $\pi(\tau)$, since $\pi$ preserves
$\tau$ isometrically. Hence $g$ is a squeezed M\"obius band of aspect ratio $\beta$.

\textbf{(b)$\Rightarrow$(a).} Given $g$ and its clean triangulation $\bar{\mathcal T}$, set
$f:=g\circ\pi:\Sg\to\R^3$, and let $\mathcal T:=\pi^{-1}(\bar{\mathcal T})$. Since $\pi$ is a
covering map and $\bar{\mathcal T}$ is finite, $\mathcal T$ is a countable, locally finite
collection of triangles tiling $\Sg$, and it is $G_\beta$-invariant because
$\pi\circ G_\beta=\pi$. Each triangle of $\mathcal T$ maps isometrically by $\pi$ onto a
triangle of $\bar{\mathcal T}$ on which $g$ is affine and squeezing, so $f$ is affine and
squeezing on each triangle of $\mathcal T$; this is property~(1). For property~(2):
$f(p)=f(q)\iff g(\pi p)=g(\pi q)\iff \pi p=\pi q$ (injectivity of $g$) $\iff p=G_\beta^k(q)$
for some $k\in\Z$ (definition of $\pi$). Thus $\beta$ is realized.
\end{proof}

By Proposition~\ref{prop:equiv}, \emph{the set of realized numbers equals the set of
aspect ratios of embedded squeezed M\"obius bands}, and we must show its infimum is $\sqrt3$.

\section{Upper bound: every $\beta>\sqrt3$ is realized}\label{sec:upper}

We construct, for each $\beta>\sqrt3$, an explicit embedded squeezed M\"obius band of
aspect ratio $\beta$. By Proposition~\ref{prop:equiv} this shows every such $\beta$ is
realized, hence $\inf\{\text{realized}\}\le\inf_{\beta>\sqrt3}\beta=\sqrt3$.

\subsection{Strategy and the squeeze criterion}

The construction is the standard \emph{triangular M\"obius band}, pushed slightly out of
the plane so as to be embedded. We first record a purely local criterion that makes the
squeeze condition easy to verify.

\begin{lemma}[Squeeze criterion]\label{lem:squeeze}
Let $\tau\subset\Sg$ be a clean triangle with base $[a,b]\times\{0\}$ (length
$\ell=b-a>0$) and apex $v=(v_1,1)$ with $v_1\in[a,b]$, and let $A:\R^2\to\R^3$ be affine,
$A(z)=A_0+Lz$, with linear part $L=[\,L_1\ \ L_2\,]$, $L_1,L_2\in\R^3$ the images of the
standard basis vectors. Assume the columns are orthogonal, $\langle L_1,L_2\rangle=0$, and
\begin{equation}\label{eq:crit}
\norm{L_1}<1,\qquad
\norm{L_2}^2>1+\ell^2\bigl(1-\norm{L_1}^2\bigr).
\end{equation}
Then $A|_\tau$ strictly shortens the base and strictly lengthens both legs. The same holds
for a clean triangle with base on $\R\times\{1\}$ and apex on $\R\times\{0\}$ with
$v_1\in[a,b]$.
\end{lemma}

\begin{proof}
The base vector is $\ell\,e_1$, whose image $\ell L_1$ has length $\ell\norm{L_1}<\ell$ by
the first inequality in \eqref{eq:crit}: the base is shortened.

A leg joins a base endpoint $(c,0)$, $c\in\{a,b\}$, to the apex $(v_1,1)$; its vector is
$(s,1)$ with $s=v_1-c$, of length $\sqrt{1+s^2}$. Since $v_1\in[a,b]$ and $c\in\{a,b\}$ we
have $\abs s=\abs{v_1-c}\le b-a=\ell$. The image leg vector is $sL_1+L_2$, and using
$\langle L_1,L_2\rangle=0$,
\[
\norm{sL_1+L_2}^2=s^2\norm{L_1}^2+\norm{L_2}^2 .
\]
We must show $\norm{sL_1+L_2}^2>1+s^2$, i.e.
$\norm{L_2}^2-1>s^2(1-\norm{L_1}^2)$. Since $\abs s\le\ell$ and $1-\norm{L_1}^2>0$,
\[
s^2\bigl(1-\norm{L_1}^2\bigr)\le\ell^2\bigl(1-\norm{L_1}^2\bigr)<\norm{L_2}^2-1,
\]
the last inequality being the second part of \eqref{eq:crit}. Hence both legs lengthen.

For a clean triangle with base on $\R\times\{1\}$, apply the isometry $y\mapsto 1-y$ of
$\Sg$, which carries it to a triangle of the first type and fixes the column norms of $L$
(it negates the $e_2$-component of the domain, replacing $L_2$ by $-L_2$, which preserves
$\norm{L_2}$ and orthogonality); the conclusion transfers.
\end{proof}

Thus it suffices to exhibit, for each $\beta>\sqrt3$, a clean triangulation of $M_\beta$ by
thin triangles ($\ell\to 0$) and a continuous \emph{injective} map, affine on each
triangle, whose linear parts have orthogonal columns $L_1,L_2$ with $\norm{L_1}<1$ and
$\norm{L_2}>1$ uniformly. We obtain such a map as an \emph{anisotropic affine reparametrization
of an isometrically embedded flat M\"obius band}, using the constructive (existence)
direction of the paper M\"obius band theory.

\subsection{The model: an isometric paper band of intermediate aspect ratio}\label{sec:model}

We use the following classical existence result, which is the elementary direction of the
optimal paper M\"obius band problem and is proved by an explicit construction (the smoothed
triangular band; see Halpern--Weaver 1977).

\begin{theorem}[Existence of paper M\"obius bands, $\lambda>\sqrt3$]\label{thm:exist}
For every $\lambda>\sqrt3$ and every $w>0$ there is a smooth embedding
$J:B_{\lambda,w}\hookrightarrow\R^3$ of the flat M\"obius band $B_{\lambda,w}$ of width $w$
and aspect ratio $\lambda$ --- i.e. the rectangle $[0,\lambda w]\times[0,w]$ with the flip
gluing $(0,t)\sim(\lambda w,\,w-t)$, carrying its flat metric --- such that $J$ is a
\emph{local isometry} (it preserves the lengths of all curves). Equivalently, in the
coordinates $(\xi,t)\in[0,\lambda w]\times[0,w]$ the differential $dJ$ has orthonormal
columns at every point.
\end{theorem}

\noindent The map $J$ exists because a flat band of aspect ratio $>\sqrt3$ can be folded
along straight creases into an embedded ``triangular'' M\"obius band, then smoothed; the
length-preserving (isometric) property is automatic for a folded developable surface. We use
$J$ as a black box.

\begin{lemma}[The squeezing reparametrization]\label{lem:embedmodel}
Fix $\beta>\sqrt3$. Choose $\lambda$ and $w$ with
\[
\sqrt3<\lambda<\beta,\qquad 1<w<\frac{\beta}{\lambda}.
\]
(Such $\lambda,w$ exist: pick any $\lambda\in(\sqrt3,\beta)$; then $\beta/\lambda>1$, so the
interval $(1,\beta/\lambda)$ for $w$ is nonempty.) Let $J:B_{\lambda,w}\hookrightarrow\R^3$
be as in Theorem~\ref{thm:exist}, and define $\Phi:M_\beta\to\R^3$ by
\begin{equation}\label{eq:Phidef}
\Phi(x,y)=J\!\left(\frac{\lambda w}{\beta}\,x,\; w\,y\right),\qquad (x,y)\in[0,\beta]\times[0,1].
\end{equation}
Then:
\begin{enumerate}
\item[(i)] $\Phi$ respects the gluing $(0,y)\sim(\beta,1-y)$ of $M_\beta$, hence is a
well-defined continuous map $M_\beta\to\R^3$;
\item[(ii)] $\Phi$ is injective;
\item[(iii)] at every smooth point the differential is $d\Phi=[\,L_1\ \ L_2\,]$ with
$L_1=\partial_x\Phi$, $L_2=\partial_y\Phi$ orthogonal, and
\[
\norm{L_1}=\frac{\lambda w}{\beta}<1,\qquad \norm{L_2}=w>1.
\]
\end{enumerate}
\end{lemma}

\begin{proof}
The affine map $R(x,y)=\big(\tfrac{\lambda w}{\beta}x,\,wy\big)$ sends $[0,\beta]\times[0,1]$
onto $[0,\lambda w]\times[0,w]$ bijectively, with $R(0,y)=(0,wy)$ and
$R(\beta,1-y)=(\lambda w,\,w(1-y))=(\lambda w,\,w-wy)$. Under the flip gluing of
$B_{\lambda,w}$, $(0,wy)\sim(\lambda w,\,w-wy)$; hence $R$ carries the gluing of $M_\beta$ to
the gluing of $B_{\lambda,w}$. Since $J$ is well defined on $B_{\lambda,w}$ (it respects that
gluing), $\Phi=J\circ R$ respects the gluing of $M_\beta$, proving (i). As $J$ is injective
(an embedding) and $R$ is a bijection compatible with the gluings, $\Phi$ is injective,
proving (ii).

For (iii): the linear part of $R$ is the diagonal matrix
$\diag\!\big(\tfrac{\lambda w}{\beta},\,w\big)$, so $\partial_x\Phi=\tfrac{\lambda
w}{\beta}\,(dJ)\,e_1$ and $\partial_y\Phi=w\,(dJ)\,e_2$, where $(dJ)e_1,(dJ)e_2$ are the
columns of $dJ$. By Theorem~\ref{thm:exist} these columns are orthonormal, so $L_1\perp L_2$,
$\norm{L_1}=\tfrac{\lambda w}{\beta}$ and $\norm{L_2}=w$. Finally $w<\beta/\lambda$ gives
$\tfrac{\lambda w}{\beta}<1$, and $w>1$ gives $\norm{L_2}=w>1$.
\end{proof}

\subsection{Discretization and conclusion of (U)}

\begin{proposition}[Upper bound]\label{prop:upper}
For every $\beta>\sqrt3$ there is an embedded squeezed M\"obius band of aspect ratio
$\beta$. Consequently every $\beta>\sqrt3$ is realized, and
$\inf\{\text{realized numbers}\}\le\sqrt3$.
\end{proposition}

\begin{proof}
Fix $\beta>\sqrt3$ and take $\Phi$, with $\lambda,w$, from Lemma~\ref{lem:embedmodel}. Choose a clean
triangulation $\bar{\mathcal T}_n$ of $M_\beta$ into thin triangles of base length
$\ell_n=\beta/n$ in the standard ``up/down'' pattern of a thin rectangle (each unit-width
sub-rectangle $[\,i\ell_n,(i+1)\ell_n\,]\times[0,1]$ is split by a diagonal into two clean
triangles, one with base on $\R\times\{0\}$ and one with base on $\R\times\{1\}$); make the
$x$-cut set symmetric under $y\mapsto1-y$ and compatible with the gluing, which is possible
since the gluing is $(0,y)\sim(\beta,1-y)$ and the cuts are at $x=i\beta/n$. In the up/down pattern, every clean triangle has its apex at a rectangle corner lying
directly above or below an endpoint of its base, so the apex abscissa satisfies $v_1\in[a,b]$
as required by Lemma~\ref{lem:squeeze}. Let $g_n$ be the affine interpolant of $\Phi$ on
$\bar{\mathcal T}_n$ (it agrees with $\Phi$ at all vertices and is affine on each triangle).
Here $\Phi=J\circ R$ is smooth (Theorem~\ref{thm:exist} gives $J\in C^\infty$ and $R$ is
affine), with bounded second derivatives on the compact $M_\beta$.

\emph{$g_n$ has the squeeze property for large $n$.} On each triangle of
$\bar{\mathcal T}_n$, the linear part $L^{(n)}$ of $g_n$ converges to the value of $d\Phi$ at
the triangle as $n\to\infty$, uniformly over all triangles, with
$\norm{L^{(n)}-d\Phi}=O(\ell_n)=O(1/n)$ (standard first-order error of affine interpolation
of a $C^2$ map; the constant depends only on $\sup\norm{D^2\Phi}$). By
Lemma~\ref{lem:embedmodel}(iii), $d\Phi$ has orthogonal columns with
$\norm{L_1}=\tfrac{\lambda w}{\beta}<1$ and $\norm{L_2}=w>1$, both constants independent of
$n$. Hence for $n$ large the columns $L_1^{(n)},L_2^{(n)}$ satisfy
\[
\norm{L_1^{(n)}}\le \tfrac12\Big(1+\tfrac{\lambda w}{\beta}\Big)<1,\qquad
\norm{L_2^{(n)}}\ge \tfrac12(1+w)>1,\qquad
\abs{\langle L_1^{(n)},L_2^{(n)}\rangle}=O(1/n).
\]
We verify the leg inequality directly (rather than via the orthogonal form of
\eqref{eq:crit}, to absorb the small non-orthogonality). For a leg parameter $\abs s\le\ell_n$,
\[
\norm{sL_1^{(n)}+L_2^{(n)}}^2
=s^2\norm{L_1^{(n)}}^2+2s\langle L_1^{(n)},L_2^{(n)}\rangle+\norm{L_2^{(n)}}^2
\ge \norm{L_2^{(n)}}^2-2\ell_n\abs{\langle L_1^{(n)},L_2^{(n)}\rangle},
\]
which tends to $w^2>1$, while $1+s^2\le 1+\ell_n^2\to1$; so for $n$ large the left side
exceeds $1+s^2$, i.e. every leg lengthens. The base shortens because
$\norm{L_1^{(n)}}<1$. Thus $g_n$ is squeezing on every triangle for all $n\ge N$.

\emph{$g_n$ is injective for large $n$.} $\Phi$ is a continuous injection of the compact
space $M_\beta$, hence a homeomorphism onto its image; in particular there is $\delta>0$ such
that $\norm{\Phi(p)-\Phi(q)}\ge\delta$ whenever the intrinsic distance
$d_{M_\beta}(p,q)\ge\tfrac1{4}$ (else a sequence of point-pairs would contradict the
homeomorphism property by compactness). Moreover $d\Phi$ has rank $2$ everywhere (orthogonal
columns of norms $\tfrac{\lambda w}{\beta}>0$ and $w>0$), so $\Phi$ is an immersion with a
uniform constant $m>0$ satisfying $\norm{\Phi(p)-\Phi(q)}\ge m\,d_{M_\beta}(p,q)$ for
$d_{M_\beta}(p,q)\le\tfrac14$ (uniform local bi-Lipschitz lower bound from the inverse
function theorem on the compact $M_\beta$). Since $\norm{g_n-\Phi}_\infty=O(1/n)\to0$, for $n$
large $\norm{g_n-\Phi}_\infty<\tfrac13\min(\delta,\tfrac{m}{4})$. Then: for
$d_{M_\beta}(p,q)\ge\tfrac14$, $\norm{g_n(p)-g_n(q)}\ge\delta-\tfrac{2}{3}\delta>0$; for
$0<d_{M_\beta}(p,q)<\tfrac14$, $g_n$'s linear parts are within $O(1/n)$ of the rank-$2$
immersion, so $g_n$ is injective on each $\tfrac14$-ball as well. Hence $g_n$ is globally
injective for all $n\ge N'$.

Take $n\ge\max(N,N')$. Then $g:=g_n$ is continuous, injective, affine on each triangle of
the clean triangulation $\bar{\mathcal T}_n$, and squeezing on each. By
Definition~\ref{def:smb} it is an embedded squeezed M\"obius band of aspect ratio $\beta$.
By Proposition~\ref{prop:equiv}, $\beta$ is realized. As $\beta>\sqrt3$ was arbitrary,
$\inf\{\text{realized}\}\le\sqrt3$.
\end{proof}

\section{Lower bound: no $\beta<\sqrt3$ is realized}\label{sec:lower}

We must show $\inf\{\text{realized}\}\ge\sqrt3$, i.e. that an embedded squeezed M\"obius band
forces $\beta\ge\sqrt3$. We first prove a clean self-contained bound, then reduce the sharp
constant to a single named theorem.

\subsection{Topological non-degeneracy and a complete bound $\beta\ge1$}

\begin{lemma}[Long generators exist]\label{lem:gen}
Let $g:M_\beta\to\R^3$ be an embedded squeezed M\"obius band with clean triangulation
$\bar{\mathcal T}$. For each triangle $\tau\in\bar{\mathcal T}$ with base $b_\tau$ on
$\bd M_\beta$ and apex on the opposite boundary segment, the image $g(\tau)$ is a genuine
(non-degenerate) Euclidean triangle whose two legs have length $>1$.
\end{lemma}

\begin{proof}
Each clean triangle $\tau$ in $M_\beta$ has its apex on $\bd M_\beta$ as well (a clean
triangle has all three vertices in $\bd M_\beta$, since by Definition~\ref{def:smb}/the
problem's definition a clean triangle is the convex hull of three boundary points). The two
legs of $\tau$ join boundary points on opposite components of the fundamental strip and so
have length $\ge1$ (the width of $\Sg$); the squeeze condition makes their images strictly
longer, hence $>1$. A degenerate image (three collinear image vertices) would force one leg
length to equal the sum or difference of the other two; but $g$ is injective and affine on
$\tau$, so $g$ restricted to $\tau$ is an affine injection of a $2$-simplex, whose image is a
non-degenerate triangle. Thus $g(\tau)$ is a non-degenerate triangle with both legs $>1$.
\end{proof}

\begin{proposition}[A complete lower bound]\label{prop:weak}
Every embedded squeezed M\"obius band has aspect ratio $\beta\ge1$. Hence
$\inf\{\text{realized}\}\ge1$.
\end{proposition}

\begin{proof}
Let $C=\bd M_\beta$ be the boundary circle. In the gluing model $C$ is obtained from the two
segments $[0,\beta]\times\{0\}$ and $[0,\beta]\times\{1\}$ joined end-to-end by the flip into
a single circle of flat length $2\beta$. Each edge of $C$ that occurs in the triangulation
$\bar{\mathcal T}$ is the base of a (unique) clean triangle, and the bases of the triangles
along the boundary tile $C$; hence
\begin{equation}\label{eq:boundary-length}
\sum_{\text{boundary edges }b}(\text{flat length of }b)=2\beta .
\end{equation}
The image $\Gamma:=g(C)$ is a Jordan curve (continuous injective image of a circle from the
compact $M_\beta$). Its length is the sum of the image base lengths, and since the squeeze
strictly shortens every base,
\begin{equation}\label{eq:Gamma-length}
\mathrm{length}(\Gamma)=\sum_{\text{boundary edges }b}\norm{g(\text{endpoints of }b)}
<\sum_b(\text{flat length of }b)=2\beta ,
\end{equation}
where the inequality is term-by-term and at least one term is strict. (We use that on each
boundary edge $g$ is affine, so $g(b)$ is a straight segment of length equal to the distance
between the images of its endpoints, which is less than the flat length of $b$.)

Now pick any clean triangle $\tau$ and one of its legs, with endpoints $p,q\in C$. By
Lemma~\ref{lem:gen} the image leg $g(p)g(q)$ is a straight segment of Euclidean length
$\norm{g(p)-g(q)}>1$. The two points $g(p),g(q)$ lie on $\Gamma$ and split it into two arcs
$\Gamma_1,\Gamma_2$ with $\mathrm{length}(\Gamma_1)+\mathrm{length}(\Gamma_2)=
\mathrm{length}(\Gamma)$. The straight-line distance between two points of a rectifiable
curve never exceeds the length of either connecting arc, so
\[
1<\norm{g(p)-g(q)}\le\min\bigl(\mathrm{length}(\Gamma_1),\mathrm{length}(\Gamma_2)\bigr)
\le\tfrac12\,\mathrm{length}(\Gamma).
\]
Combining with \eqref{eq:Gamma-length}, $1<\tfrac12\,\mathrm{length}(\Gamma)<\tfrac12\cdot
2\beta=\beta$. Hence $\beta>1$, and so $\inf\{\text{realized}\}\ge1$.
\end{proof}

\begin{remark}
Proposition~\ref{prop:weak} is fully rigorous and unconditional but far from the sharp value
$\sqrt3$. The chord-versus-arc argument uses only the length of a single leg and the total
boundary length; it ignores the \emph{angular} interaction between the two legs of a triangle
and the way the half-twist forces near-coplanar generators to be nearly perpendicular ---
exactly the mechanism producing the sharp constant, addressed next.
\end{remark}

\subsection{The sharp bound via the Halpern--Weaver inequality}

The sharp constant is governed by a metric inequality for embedded M\"obius bands made of a
flat strip; the squeeze condition is precisely the polyhedral form of ``built from a flat
strip with legs at least as long as in the flat model''. We isolate the needed theorem.

\begin{theorem}[Optimal paper M\"obius band; resolution of the Halpern--Weaver
conjecture, Schwartz 2025]\label{thm:HW}
There is no smooth embedded M\"obius band isometric to a flat rectangle of aspect ratio
$\lambda$ (length $\lambda$, width $1$, glued by a flip) with $\lambda\le\sqrt3$. Moreover
the infimum of admissible $\lambda$ equals $\sqrt3$, and minimizing sequences converge to the
doubly-covered equilateral triangle.
\end{theorem}

\begin{proposition}[Sharp lower bound]\label{prop:sharp}
Every embedded squeezed M\"obius band has aspect ratio $\beta\ge\sqrt3$. Hence
$\inf\{\text{realized}\}\ge\sqrt3$.
\end{proposition}

\begin{proof}[Proof, conditional on Theorem~\ref{thm:HW} and Lemma~\ref{lem:mono} below]
Let $g:M_\beta\to\R^3$ be an embedded squeezed M\"obius band. By Lemma~\ref{lem:mono} there
is a smooth embedded M\"obius band isometric to a flat rectangle of aspect ratio
$\lambda\le\beta$. If $\beta<\sqrt3$ then $\lambda<\sqrt3$, contradicting
Theorem~\ref{thm:HW}. Hence $\beta\ge\sqrt3$, and $\inf\{\text{realized}\}\ge\sqrt3$.
\end{proof}

We now state precisely the one geometric lemma that completes the sharp bound, and explain
honestly its status.

\begin{lemma}[Aspect-ratio monotonicity under squeezing]\label{lem:mono}
Let $g:M_\beta\to\R^3$ be an embedded squeezed M\"obius band. Then there is a smooth
embedded M\"obius band in $\R^3$ that is isometric (as a surface with its induced path
metric) to a flat rectangle of aspect ratio $\lambda$ for some $\lambda\le\beta$.
\end{lemma}

\noindent\emph{Status of Lemma~\ref{lem:mono}.}
The intended mechanism is as follows. The image $\hat M:=g(M_\beta)$, with the path metric
induced from $\R^3$, is a piecewise-Euclidean M\"obius band: each triangle $g(\tau)$ is a flat
Euclidean triangle, and these are glued along shared edges, with possible cone points at
vertices. The squeeze condition asserts, triangle by triangle, that every boundary (base)
edge is \emph{shorter} and every transverse (leg) edge is \emph{longer} than in the flat
domain $M_\beta$. Morally this means $\hat M$ is ``shorter along its boundary circle and
wider across'' than $M_\beta$, so the natural aspect ratio (boundary length over width) of
$\hat M$ does not exceed $\beta$; developing $\hat M$ onto a flat rectangle and smoothing the
cone points (all cone angles are $\le2\pi$ for an embedded surface) would then give a smooth
embedded paper M\"obius band of aspect ratio $\lambda\le\beta$. \emph{Making this rigorous ---
defining the right notion of aspect ratio for the cone-metric band, proving the edgewise
inequalities aggregate to a global inequality on aspect ratios, and controlling the smoothing
so that embeddedness and the aspect-ratio bound survive --- is exactly the polyhedral content
of Schwartz's theorem and is not carried out here.} This is the single missing ingredient: we
have reduced the sharp lower bound to Theorem~\ref{thm:HW} together with
Lemma~\ref{lem:mono}, and we do not claim a complete proof of Lemma~\ref{lem:mono}.

\begin{remark}[What is complete]\label{rem:honest}
Unconditionally and completely, we have proved: the reduction (Proposition~\ref{prop:equiv}),
the full upper bound $\inf\{\text{realized}\}\le\sqrt3$ via explicit construction
(Proposition~\ref{prop:upper}, modulo only the cited classical existence
Theorem~\ref{thm:exist} of paper bands for $\lambda>\sqrt3$), and the lower bound
$\inf\{\text{realized}\}\ge1$ (Proposition~\ref{prop:weak}). The sharp lower bound
$\inf\{\text{realized}\}\ge\sqrt3$ is established \emph{conditionally} on Theorem~\ref{thm:HW}
and Lemma~\ref{lem:mono}. Thus the answer $\sqrt3$ is correct and is proved on the upper side;
the lower side is reduced to one deep theorem and one geometric monotonicity lemma.
\end{remark}

\section{Conclusion}

\begin{theorem}\label{thm:main}
The infimum of the set of realized numbers is $\sqrt3$.
\end{theorem}

\begin{proof}
By Proposition~\ref{prop:upper}, every $\beta>\sqrt3$ is realized, so
$\inf\{\text{realized}\}\le\sqrt3$; this is complete and unconditional. By
Proposition~\ref{prop:sharp} (using Theorem~\ref{thm:HW} and Lemma~\ref{lem:mono}), every
realized $\beta$ satisfies $\beta\ge\sqrt3$, so $\inf\{\text{realized}\}\ge\sqrt3$. Combining,
$\inf\{\text{realized}\}=\sqrt3$. Unconditionally we have the two-sided bound
$1\le\inf\{\text{realized}\}\le\sqrt3$ (Propositions~\ref{prop:weak} and \ref{prop:upper}).
\end{proof}
